$$
\begin{array}{l} \zeta_ {i} = c _ {i} \left(\sum_ {j \in \mathcal {N} _ {i}} a _ {i j} \left(\phi_ {j} - \phi_ {i}\right) + g _ {i} \left(V _ {\text { ref }} - \phi_ {i}\right) \right. \\ \left. + \sum_ {j \in \mathcal {N} _ {i}} a _ {i j} \left(R _ {j} ^ {\text {vir}} \psi_ {j} - R _ {i} ^ {\text {vir}} \psi_ {i}\right)\right) \tag {23} \\ = c _ {i} \left(\sum_ {j \in \mathcal {N} _ {i}} a _ {i j} \left(\Phi_ {j} - \Phi_ {i}\right) + g _ {i} \left(\Theta_ {\text { ref }} - \Phi_ {i}\right)\right) \\ = - c _ {i} \left(d _ {i} + g _ {i}\right) \Phi_ {i} + c _ {i} \sum_ {j \in \mathcal {N} _ {i}} a _ {i j} \Phi_ {j} + c _ {i} g _ {i} \Theta_ {\mathrm{ref}}. \\ \end{array}
