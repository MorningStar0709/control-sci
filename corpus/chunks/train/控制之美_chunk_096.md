$$
\begin{array}{l} \omega_ {n} ^ {2} = A \left(0 - s _ {p 2}\right) \left(0 - s _ {p 3}\right) \\ = A \left(\zeta \omega_ {\mathrm{n}} - \omega_ {\mathrm{n}} \sqrt {1 - \zeta^ {2}} \mathrm{j}\right) \left(\zeta \omega_ {\mathrm{n}} + \omega_ {\mathrm{n}} \sqrt {1 - \zeta^ {2}} \mathrm{j}\right) = A \omega_ {\mathrm{n}} ^ {2} \\ \Rightarrow A = 1 \tag {5.3.6a} \\ \end{array}
$$

令 $s = s_{\mathrm{p2}} = -\zeta \omega_{\mathrm{n}} + \omega_{\mathrm{n}}\sqrt{1 - \zeta^2}\mathrm{j},$ 可得

$$
\begin{array}{l} \omega_ {n} ^ {2} = B \left(- \zeta \omega_ {n} + \omega_ {n} \sqrt {1 - \zeta^ {2}} j - s _ {p 1}\right) \left(- \zeta \omega_ {n} + \omega_ {n} \sqrt {1 - \zeta^ {2}} j - s _ {p 3}\right) \\ = B \left(- \zeta \omega_ {\mathrm{n}} + \omega_ {\mathrm{n}} \sqrt {1 - \zeta^ {2}} \mathrm{j} - 0\right) \left[ - \zeta \omega_ {\mathrm{n}} + \omega_ {\mathrm{n}} \sqrt {1 - \zeta^ {2}} \mathrm{j} - \left(- \zeta \omega_ {\mathrm{n}} - \omega_ {\mathrm{n}} \sqrt {1 - \zeta^ {2}} \mathrm{j}\right) \right] \\ = B \left(- \zeta \omega_ {\mathrm{n}} + \omega_ {\mathrm{n}} \sqrt {1 - \zeta^ {2}} \mathrm{j}\right) \left(2 \omega_ {\mathrm{n}} \sqrt {1 - \zeta^ {2}} \mathrm{j}\right) \\ = B \left[ - 2 \zeta \omega_ {\mathrm{n}} ^ {2} \sqrt {1 - \zeta^ {2}} \mathrm{j} - 2 \omega_ {\mathrm{n}} ^ {2} (1 - \zeta^ {2}) \right] \\ = B \left(- 2 \omega_ {\mathrm{n}} ^ {2}\right) \left(\zeta \sqrt {1 - \zeta^ {2}} \mathrm{j} + 1 - \zeta^ {2}\right) \\ \end{array}
\Rightarrow B = \left(- \frac {1}{2}\right) \frac {1}{\zeta \sqrt {1 - \zeta^ {2}} j + 1 - \zeta^ {2}}= \left(- \frac {1}{2}\right) \frac {1}{\zeta \sqrt {1 - \zeta^ {2}} j + 1 - \zeta^ {2}} \frac {\left(- \zeta \sqrt {1 - \zeta^ {2}} j\right) + 1 - \zeta^ {2}}{\left(- \zeta \sqrt {1 - \zeta^ {2}} j\right) + 1 - \zeta^ {2}}= \left(- \frac {1}{2}\right) \frac {1 - \zeta^ {2} - \zeta \sqrt {1 - \zeta^ {2}} j}{(1 - \zeta^ {2}) ^ {2} + \zeta^ {2} (1 - \zeta^ {2})} = \left(- \frac {1}{2}\right) \frac {1 - \zeta^ {2} - \zeta \sqrt {1 - \zeta^ {2}} j}{1 - 2 \zeta^ {2} + \zeta^ {4} + \zeta^ {2} - \zeta^ {4}}= \left(- \frac {1}{2}\right) \frac {1 - \zeta^ {2} - \zeta \sqrt {1 - \zeta^ {2}} j}{1 - \zeta^ {2}}= - \frac {1}{2} \left(1 - \frac {\zeta}{\sqrt {1 - \zeta^ {2}}} j\right) \tag {5.3.6b}
$$

$$C = - \frac {1}{2} \left(1 + \frac {\zeta}{\sqrt {1 - \zeta^ {2}}} \mathrm{j}\right) \tag {5.3.6c}$$
