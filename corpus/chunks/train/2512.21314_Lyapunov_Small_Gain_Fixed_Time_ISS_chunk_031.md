$$
\begin{array}{l} \mathcal {G} _ {\theta} (\tilde {x} + h (\varphi (\theta)), \tilde {u} + \varphi (\theta)) \\ = \mathcal {G} _ {\theta} (h (\tilde {u} + \varphi (\theta)), \tilde {u} + \varphi (\theta)) \\ + \mathcal {G} _ {\theta} (\tilde {x} + h (\varphi (\theta)), \tilde {u} + \varphi (\theta)) - \mathcal {G} _ {\theta} (h (\tilde {u} + \varphi (\theta)), \tilde {u} + \varphi (\theta)), \tag {34} \\ \end{array}
$$

it follows via Assumptions 3 and 6 that

$$
\begin{array}{l} \left| \mathcal {G} _ {\theta} (\tilde {x} + h (\varphi (\theta)), \tilde {u} + \varphi (\theta)) \right| \leq \left| \mathcal {G} _ {\theta} (h (\tilde {u} + \varphi (\theta)), \tilde {u} + \varphi (\theta)) \right| \\ + K (| \tilde {x} | + \ell | \tilde {u} |). \tag {35} \\ \end{array}
$$

Moreover, by Assumption 5, we have

$$\mu | \tilde {u} | \leq | \mathcal {G} _ {\theta} (h (\tilde {u} + \varphi (\theta)), \tilde {u} + \varphi (\theta)) | \leq L | \tilde {u} |. \tag {36}$$

From (30), we know that $\mu - \ell K > 0 .$ , which implies that we can pick $\delta \in ( 0 , \mu - \ell K )$ sufficiently small, such that $\gamma \gamma _ { \delta } \in ( 0 , 1 )$ , where

$$\gamma_ {\delta} := \frac {K}{\mu - \ell K - \delta}.$$

Suppose that $| \tilde { u } | \geq \gamma _ { \delta } | \tilde { x } | .$ . Then, using (36), we can upperbound (35) as follows:

$$
\begin{array}{l} \left. \left| \mathcal {G} _ {\theta} (\tilde {x} + h (\varphi (\theta)), \tilde {u} + \varphi (\theta)) \right| \leq \left(L + K \left(\frac {1}{\gamma_ {\delta}} + \ell\right)\right) | \tilde {u} | \right. \\ = (L + \mu - \delta) | \tilde {u} |. \tag {37} \\ \end{array}
$$

Similarly, using again (34), Assumptions 3 and $^ { 6 , }$ and (36), we have:

$$
\begin{array}{l} \left| \mathcal {G} _ {\theta} (\tilde {x} + h (\varphi (\theta)), \tilde {u} + \varphi (\theta)) \right| \\ \geq \left| \mathcal {G} _ {\theta} (h (\tilde {u} + \varphi (\theta)), \tilde {u} + \varphi (\theta)) \right| - K (| \tilde {x} | + \ell | \tilde {u} |) \\ \geq \left(\mu - K \left(\frac {1}{\gamma_ {\delta}} + \ell\right)\right) | \tilde {u} | \\ = \delta | \tilde {u} |, \tag {38} \\ \end{array}
$$

for all $| \tilde { u } | \geq \gamma _ { \delta } | \tilde { x } |$ , and we also obtain:

$$
\begin{array}{l} \tilde {u} ^ {\top} \mathcal {G} _ {\theta} (\tilde {x} + h (\varphi (\theta)), \tilde {u} + \varphi (\theta)) \geq \mu | \tilde {u} | ^ {2} - K | \tilde {u} | (| \tilde {x} | + \ell | \tilde {u} |) \\ \geq \delta | \tilde {u} | ^ {2}. \tag {39} \\ \end{array}
$$
