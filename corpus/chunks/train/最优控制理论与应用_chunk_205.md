$$\boldsymbol {u} ^ {*} (t) = - \frac {3 \left[ \tilde {\boldsymbol {x}} - \tilde {\boldsymbol {v}} \frac {\langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {x}} \rangle}{\langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {v}} \rangle} \right]}{\left(1 - \frac {c _ {\mathrm{p}}}{c _ {\mathrm{e}}}\right) \frac {\langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {x}} \rangle^ {2}}{\langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {v}} \rangle^ {2}}}= \frac {3 [ \tilde {\boldsymbol {v}} \langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {x}} \rangle - \tilde {\boldsymbol {x}} \langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {v}} \rangle ]}{\left(1 - \frac {c _ {\mathrm{p}}}{c _ {\mathrm{e}}}\right) \langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {x}} \rangle} \frac {\langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {v}} \rangle}{\langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {x}} \rangle} \tag {10-121}$$

利用三维向量的多重积公式

$$(a \times b) \times c = \langle a, c \rangle b - \langle b, c \rangle a$$

有

$$\tilde {\boldsymbol {v}} \langle \tilde {\boldsymbol {x}}, \tilde {\boldsymbol {x}} \rangle - \tilde {\boldsymbol {x}} \langle \tilde {\boldsymbol {x}}, \tilde {\boldsymbol {v}} \rangle = (\tilde {\boldsymbol {x}} \times \tilde {\boldsymbol {v}}) \times \tilde {\boldsymbol {x}} \tag {10-122}$$

于是

$$\boldsymbol {u} ^ {*} (t) = \frac {3 (\tilde {\boldsymbol {x}} \times \tilde {\boldsymbol {v}}) \times \tilde {\boldsymbol {x}}}{\left(1 - \frac {c _ {\mathrm{p}}}{c _ {\mathrm{e}}}\right) \langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {x}} \rangle} \frac {\langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {v}} \rangle}{\langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {x}} \rangle} \tag {10-123}$$

$$\dot {\boldsymbol {q}} = \frac {(\tilde {\boldsymbol {x}} \times \tilde {\boldsymbol {v}})}{\langle \tilde {\boldsymbol {x}} , \tilde {\boldsymbol {x}} \rangle} \tag {10-124}$$
