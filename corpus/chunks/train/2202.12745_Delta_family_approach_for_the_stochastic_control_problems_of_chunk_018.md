$$V _ {m _ {1} m _ {2}} ^ {n} \approx V _ {m _ {1} m _ {2}} ^ {n + 1} + h \sum_ {k _ {1} = 0} ^ {M _ {1}} \sum_ {k _ {2} = 0} ^ {M _ {2}} V _ {k _ {1} k _ {2}} ^ {n + 1} \int_ {\Omega^ {2}} g _ {m _ {1}} ^ {(1)} (y _ {1}) g _ {m _ {2}} ^ {(2)} (y _ {2}) \mathcal {L} ^ {\pi_ {\mathbf {n} + 1} ^ {*}} \left(g _ {k _ {1}} ^ {(1)} (y _ {1}) g _ {k _ {2}} ^ {(2)} (y _ {2})\right) d \mathbf {y}, \tag {3.11}$$

and terminal condition

$$V _ {m _ {1} m _ {2}} ^ {N} = \int_ {\Omega^ {2}} U (\hat {x} (y _ {1})) g _ {m _ {1}} ^ {(1)} (\hat {x} (y _ {1})) g _ {m _ {2}} ^ {(2)} (\hat {v} (y _ {2})) d \mathbf {y}, \quad m _ {1} = 0, 1, \dots , M _ {1}; m _ {2} = 0, 1, \dots , M _ {2}. \tag {3.12}$$

$$V (t _ {n}, x, v) \approx \sum_ {m _ {1} = 0} ^ {M _ {1}} \sum_ {m _ {2} = 0} ^ {M _ {2}} V _ {m _ {1} m _ {2}} ^ {n} g _ {m _ {1}} ^ {(1)} (\hat {x} (x)) g _ {m _ {2}} ^ {(2)} (\hat {v} (v)),$$

$$\pi^ {*} (t _ {n}, x, v) \approx \pi_ {n} ^ {*} (x, v).$$
