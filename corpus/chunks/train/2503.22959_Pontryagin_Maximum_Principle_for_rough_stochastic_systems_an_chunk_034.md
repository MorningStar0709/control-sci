$$
\begin{array}{l} \delta (Z ^ {\prime} - \bar {Z} ^ {\prime}) _ {s, t} = \delta (\tilde {F} X) _ {s, t} ^ {\prime} + \delta (F \tilde {X}) _ {s, t} ^ {\prime} + \delta \widetilde {f} _ {s, t}. \\ = \delta (\tilde {F} ^ {\prime} X) _ {s, t} + \delta (\tilde {F} X ^ {\prime}) _ {s, t} + \delta (F ^ {\prime} \tilde {X}) _ {s, t} + \delta (F \tilde {X} ^ {\prime}) _ {s, t} + \delta \tilde {f} _ {s, t} ^ {\prime}, \\ \end{array}
$$

where we set ${ \tilde { F } } : = F - { \bar { F } }$ and similar for $\tilde { F } ^ { \prime } , \tilde { f } ^ { \prime } .$ . Since $\tilde { X } ^ { \prime } = \tilde { F } X + F \tilde { X } + \tilde { f }$ it follows that

$$
\begin{array}{l} \left\| \delta Z _ {s, t} ^ {\prime} - \delta \bar {Z} _ {s, t} ^ {\prime} \right\| _ {m} \\ \lesssim \| \delta \tilde {F} _ {s, t} ^ {\prime} \| _ {\infty} \| X _ {t} \| _ {m} + \| \tilde {F} _ {s} ^ {\prime} \| _ {\infty} \| \delta X _ {s, t} \| _ {m} + \| \delta \tilde {F} _ {s, t} \| _ {\infty} \| X _ {t} ^ {\prime} \| _ {m} + \| \tilde {F} _ {s} \| _ {\infty} \| \delta X _ {s, t} ^ {\prime} \| _ {m} \tag {3.47} \\ + \| \delta F _ {s, t} ^ {\prime} \| _ {\infty} \| \tilde {X} _ {t} \| _ {m} + \| F _ {s} ^ {\prime} \| _ {\infty} \| \delta \tilde {X} _ {s, t} \| _ {m} + \| \delta F _ {s, t} \| _ {\infty} \| \tilde {X} _ {t} ^ {\prime} \| _ {m} + \| F _ {s} \| _ {\infty} \| \delta \tilde {X} _ {s, t} ^ {\prime} \| _ {m} + \| \delta \widetilde {f} _ {s, t} ^ {\prime} \| _ {m} \\ \lesssim \| \delta \tilde {X} _ {s, t} \| _ {m} + (\Gamma_ {s, t} + \theta) (t - s) ^ {\beta^ {\prime}}. \\ \end{array}
$$

Using (3.43), (3.44), (3.45), (3.46), (3.47) and (3.40), we get that

$$\| \delta J _ {s, u, t} \| _ {m} \lesssim \| J _ {s, u} \| _ {m} (t - s) ^ {\alpha} + (\Gamma_ {s, t} + \theta) (t - s) ^ {\alpha + \beta}$$

and that

$$\| \mathbb {E} _ {s} \delta J _ {s, u, t} \| _ {m} \lesssim \| J _ {s, u} \| _ {m} (t - s) ^ {\alpha} + (\Gamma_ {s, t} + \theta) (t - s) ^ {\alpha + \beta + \beta^ {\prime}}.$$

On the other hand,

$$J _ {s, t} = \left[ \int_ {s} ^ {t} Z _ {r} d \pmb {\eta} _ {r} - (Z _ {s} \delta \eta_ {s, t} + Z _ {s} ^ {\prime} \eta_ {s, t} ^ {(2)}) \right] - \left[ \int_ {s} ^ {t} \bar {Z} _ {r} d \bar {\pmb {\eta}} _ {r} - (\bar {Z} _ {s} \delta \bar {\eta} _ {s, t} + \bar {Z} _ {s} ^ {\prime} \bar {\eta} _ {s, t} ^ {(2)}) \right].$$

By the Step 3 in the proof of Proposition 3.4, we have

$$\| J _ {s, t} \| _ {m} \lesssim | t - s | ^ {\alpha + \beta}, \| \mathbb {E} _ {s} J _ {s, t} \| _ {m} \lesssim | t - s | ^ {\alpha + \beta + \beta^ {\prime}}.$$
