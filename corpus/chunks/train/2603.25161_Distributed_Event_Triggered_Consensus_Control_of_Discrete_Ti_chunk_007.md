$$
\begin{array}{l} J _ {\text { all }} (x [ 0 ]) = \sum_ {k = 0} ^ {\infty} x ^ {\top} [ k ] \left(L \otimes Q + c ^ {2} L ^ {2} \otimes F ^ {\top} R F\right) x [ k ] \\ = \sum_ {k = 0} ^ {\infty} \tilde {x} ^ {\top} [ k ] (\Lambda \otimes Q + c ^ {2} \Lambda^ {2} \otimes F ^ {\top} R F) \tilde {x} [ k ] \\ = \sum_ {k = 0} ^ {\infty} \sum_ {i = 2} ^ {N} \tilde {x} _ {i} ^ {\top} [ k ] (\lambda_ {i} (L) Q \\ \left. + c ^ {2} \lambda_ {i} (L) ^ {2} F ^ {\top} R F\right) \tilde {x} _ {i} [ k ] \\ = \sum_ {i = 2} ^ {N} \sum_ {k = 0} ^ {\infty} \tilde {x} _ {i} ^ {\top} [ k ] W _ {i} \tilde {x} _ {i} [ k ], \\ \end{array}

$$
\begin{array}{l} \tilde {x} _ {i} ^ {\top} [ k ] W _ {i} \tilde {x} _ {i} [ k ] = \tilde {x} _ {i} ^ {\top} [ k ] \left(P _ {i} - \tilde {A} _ {i} ^ {\top} P _ {i} \tilde {A} _ {i}\right) \tilde {x} _ {i} [ k ] \\ = \tilde {x} _ {i} ^ {\top} [ k ] P _ {i} \tilde {x} _ {i} [ k ] - \tilde {x} _ {i} ^ {\top} [ k + 1 ] P _ {i} \tilde {x} _ {i} [ k + 1 ]. \\ \end{array}

$$
\begin{array}{l} \sum_ {k = 0} ^ {K} \tilde {x} _ {i} ^ {\top} [ k ] W _ {i} \tilde {x} _ {i} [ k ] \\ = \sum_ {k = 0} ^ {K} \left(\tilde {x} _ {i} ^ {\top} [ k ] P _ {i} \tilde {x} _ {i} [ k ] - \tilde {x} _ {i} ^ {\top} [ k + 1 ] P _ {i} \tilde {x} _ {i} [ k + 1 ]\right) \\ = \tilde {x} _ {i} ^ {\top} [ 0 ] P _ {i} \tilde {x} _ {i} [ 0 ] - \tilde {x} _ {i} ^ {\top} [ K + 1 ] P _ {i} \tilde {x} _ {i} [ K + 1 ]. \\ \end{array}

$$\sum_ {k = 0} ^ {\infty} \tilde {x} _ {i} ^ {\top} [ k ] W _ {i} \tilde {x} _ {i} [ k ] = \tilde {x} _ {i} ^ {\top} [ 0 ] P _ {i} \tilde {x} _ {i} [ 0 ],$$
