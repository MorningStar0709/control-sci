$$
\begin{array}{l} \left| U _ {N} (z, \mu) - U (z, \mu) \right| = \left| U \left(x _ {i - 1 / 2}, \mu\right) - U (z, \mu) \right| \\ = \left| g \left(\sum_ {j = 1} ^ {N} f \left(x _ {i - 1 / 2}, x _ {j - 1 / 2}\right) \mu (t, \Omega_ {j})\right) - g \left(\int_ {\Omega} f (z, y) \mu (t, d y)\right) \right| \\ \leq L _ {g} \left| \sum_ {j = 1} ^ {N} f \left(x _ {i - 1 / 2}, x _ {j - 1 / 2}\right) \mu (t, \Omega_ {j}) - \int_ {\Omega} f (z, y) \mu (t, d y) \right| \\ = L _ {g} \left| \sum_ {j = 1} ^ {N} f \left(x _ {i - 1 / 2}, x _ {j - 1 / 2}\right) \mu (t, \Omega_ {j}) - \sum_ {j = 1} ^ {N} \int_ {\Omega_ {j}} f (z, y) \mu (t, d y) \right| \\ \leq L _ {g} \sum_ {j = 1} ^ {N} \left| f \left(x _ {i - 1 / 2}, x _ {j - 1 / 2}\right) \mu (t, \Omega_ {j}) - \int_ {\Omega_ {j}} f (z, y) \mu (t, d y) \right| \\ = L _ {g} \sum_ {j = 1} ^ {N} \left| \int_ {\Omega_ {j}} f _ {N} (x _ {i - 1 / 2}, x _ {j - 1 / 2}) \mu (t, \mathrm{d} y) - \int_ {\Omega_ {j}} f (z, y) \mu (t, \mathrm{d} y) \right| \\ \leq L _ {g} \sum_ {j = 1} ^ {N} \int_ {\Omega_ {j}} \left| f _ {N} \left(x _ {i - 1 / 2}, x _ {j - 1 / 2}\right) - f (z, y) \right| \mu (t, d y) \\ \leq L _ {g} \sum_ {j = 1} ^ {N} \mu (t, \Omega_ {j}) \max _ {y \in \Omega_ {j}, z \in \Omega_ {i}} \left| f _ {N} (x _ {i - 1 / 2}, x _ {j - 1 / 2}) - f (z, y) \right| \\ \leq L _ {g} \max _ {u, v \in \Omega} \left| f _ {N} (u, v) - f (u, v) \right| \sum_ {j = 1} ^ {N} \mu (t, \Omega_ {j}) \\ \leq L _ {g} \max _ {u, v \in \Omega} \left| f _ {N} (u, v) - f (u, v) \right| \tag {101} \\ \end{array}
$$

where we used $\sum _ { j = 1 } ^ { N } \mu \big ( t , \Omega _ { j } \big ) = 1$ . The same estimate applies to the last term of (100), yielding

$$\sup _ {x, y \in \Omega} | a (z, x, \mu) - a _ {N} (z, x, \mu) | \leq \frac {2 L _ {a} L _ {g}}{\eta} \max _ {u, v \in \Omega} | f _ {N} (u, v) - f (u, v) | \tag {102}$$

and hence

$$\int_ {A} \left(\int_ {\Omega} (a (z, x, \mu) - a _ {N} (z, x, \mu)) \mu (t, \mathrm{dz})\right) \mathrm{d} x \leq \frac {2 L _ {a} L _ {g}}{\eta} \max _ {u, v \in \Omega} | f _ {N} (u, v) - f (u, v) |. \tag {103}$$

For the last term of (98), we have

$$- \int_ {A} \left(\int_ {\Omega} (a (x, y, \mu) - a _ {N} (x, y, \mu)) \mathrm{d} y\right) \mu (t, \mathrm{d} x) \leq \int_ {A} \int_ {\Omega} | a (x, y, \mu) - a _ {N} (x, y, \mu) | \mathrm{d} y \mu (t, \mathrm{d} x). \tag {104}$$

We again use (100)–(101), and thus (104) leads to

$$- \int_ {A} \left(\int_ {\Omega} (a (x, y, \mu) - a _ {N} (x, y, \mu)) \mathrm{d} y\right) \mu (t, \mathrm{d} x) \leq \frac {2 L _ {a} L _ {g}}{\eta} \max _ {u, v \in \Omega} | f _ {N} (u, v) - f (u, v) |. \tag {105}$$
