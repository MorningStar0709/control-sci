$$
\begin{array}{l} \hat {e} ^ {P} (t _ {f} - t) = \\ \left(W _ {2 1} + W _ {2 2} V (t _ {f} - t)\right) \left(W _ {1 1} + W _ {1 2} V (t _ {f} - t)\right) ^ {- 1} \\ - W _ {2 1} W _ {1 1} ^ {- 1}. \tag {64} \\ \end{array}

$$

$$\max _ {t \in [ 0, \bar {t} ]} \| \hat {e} ^ {P} (t _ {f} - t) \| _ {\infty} = O \big (\exp (- l _ {2} t _ {f}) \big) \tag {65}$$
