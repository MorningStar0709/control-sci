```matlab
Ka=[10 80];T=[4 2];
for i=1:1:2
    G1=tf([5000],[1 1000]); % 线圈模型
    G2=zpk([],[0 -20],1); % 负载模型
    G=Ka(i)*series(G1,G2);
    sys=feedback(G,1); % 单位阶跃响应
    t=0:0.005:T(i);
    figure(i);step(sys,t);grid
end
    figure(3);sysn=-feedback(G2,Ka(2)*G1);
```
