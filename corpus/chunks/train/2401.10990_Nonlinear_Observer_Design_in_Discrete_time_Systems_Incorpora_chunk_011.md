$$\mathbf {N L} = \mathbb {U} ^ {\top} (\mathbb {H} \Phi) + \Phi^ {\top} \mathbb {H} ^ {\top} \mathbb {U} + \mathbb {M} ^ {\top} (\mathbb {G} \Psi) + \Psi^ {\top} \mathbb {G} ^ {\top} \mathbb {M}, \tag {35}$$

$$
\mathbb {U} = \left[ \begin{array}{c c c c c c c} \mathbb {U} _ {1 1} ^ {\top} & \dots & \mathbb {U} _ {1 \bar {n}} ^ {\top} & \dots & \mathbb {U} _ {m 1} ^ {\top} & \dots & \mathbb {U} _ {m \bar {n}} ^ {\top} \end{array} \right] ^ {\top}, \tag {36}
