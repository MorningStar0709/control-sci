$$\left\| x (\cdot) \right\| _ {C _ {\mathbb {A}} ([ t _ {0}, T ]; L ^ {2} (\mathscr {C}))} \leq \mathcal {C} \left\{1 + \left\| x _ {0} \right\| _ {2} + \left\| u \right\| _ {L ^ {2} (t _ {0}, T; \mathscr {H})} \right\}.$$

Assume that $\bar { u } ( \cdot ) \in \mathcal { U } ^ { 4 } [ 0 , T ]$ is an optimal control and $\bar { x } ( \cdot )$ is corresponding optimal state. Let us introduce some notations. For any $\Phi = D , F , G , L$ , write

$$
\left\{ \begin{array}{l} \Phi_ {x} (t) := \Phi_ {x} (t, \bar {x} (t), \bar {u} (t)), \\ \Phi_ {u} (t) := \Phi_ {u} (t, \bar {x} (t), \bar {u} (t)), \\ \Phi_ {x x} (t) := \Phi_ {x x} (t, \bar {x} (t), \bar {u} (t)), \\ \Phi_ {u u} (t) := \Phi_ {u u} (t, \bar {x} (t), \bar {u} (t)), \\ \Phi_ {x u} (t) := \Phi_ {x u} (t, \bar {x} (t), \bar {u} (t)). \end{array} \right.
$$

To establish the second order necessary condition for Problem (QOP), we introduce the following adjoint equations:

$$
\left\{ \begin{array}{l} d y (t) = - \left\{D _ {x} (t) ^ {*} y (t) + \left(F _ {x} (t) + \Upsilon G _ {x} (t)\right) ^ {*} Y (t) - L _ {x} (t) \right\} d t + Y (t) d W (t), \text {   in   } [ 0, T), \\ y (T) = - g _ {x} (\overline {{x}} (T)), \end{array} \right. \tag {2.1}
$$

and

$$
\left\{ \begin{array}{l} d P (t) = - \Big \{D _ {x} (t) ^ {*} P (t) + P (t) D _ {x} (t) + (F _ {x} (t) + \Upsilon G _ {x} (t)) ^ {*} Q (t) \Upsilon + Q (t) \Upsilon (F _ {x} (t) + \Upsilon G _ {x} (t)) \\ \qquad + (F _ {x} (t) + \Upsilon G _ {x} (t)) ^ {*} P (t) (F _ {x} (t) + \Upsilon G _ {x} (t)) + \mathbb {H} _ {x x} (t) \Big \} d t + Q (t) d W (t), \text {in [0,T),} \\ P (T) = - g _ {x x} (\overline {{x}} (T)), \end{array} \right. \tag {2.2}
$$

where $y ( T ) \in L ^ { 2 } ( \mathcal { C } _ { T } ) , P ( T ) \in \mathcal { L } ( L ^ { 2 } ( \mathcal { C } _ { T } ) )$ . The Hamilton function $\mathbb { H } ( \cdot , \cdot , \cdot , \cdot )$ is defined by

$$\mathbb {H} (t, x, u, y, Y) := \langle y, D (t, x, u) \rangle + \langle Y, F (t, x, u) + \Upsilon G (t, x, u) \rangle - L (t, x, u),(t, x, u, y, Y) \in [ 0, T ] \times L ^ {2} (\mathcal {C}) \times U \times L ^ {2} (\mathcal {C}) \times L ^ {2} (\mathcal {C}),$$

and simplify the notations

$$
\left\{ \begin{array}{l} \mathbb {H} _ {x} (t) := \mathbb {H} _ {x} (t, x, u, y, Y), \\ \mathbb {H} _ {u} (t) := \mathbb {H} _ {u} (t, x, u, y, Y), \\ \mathbb {H} _ {x x} (t) := \mathbb {H} _ {x x} (t, x, u, y, Y), \\ \mathbb {H} _ {u u} (t) := \mathbb {H} _ {u u} (t, x, u, y, Y), \\ \mathbb {H} _ {x u} (t) := \mathbb {H} _ {x u} (t, x, u, y, Y). \end{array} \right.
$$
