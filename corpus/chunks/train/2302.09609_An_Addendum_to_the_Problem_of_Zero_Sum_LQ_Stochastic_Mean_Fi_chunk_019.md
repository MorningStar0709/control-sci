$$
\begin{array}{l} \mathcal {J} (s, T, x _ {s}; u _ {1} (\cdot), u _ {2} (\cdot)) = \frac {1}{2} \mathbb {E} [ \langle \hat {X} (s) x _ {s}, x _ {s} \rangle ] \\ + \frac {1}{2} \mathbb {E} \left[ \int_ {s} ^ {T} \left(| \mathbb {V} _ {2 1} (t) (u _ {1} ^ {1} (t) - F _ {1} (t) x _ {u} ^ {1} (t)) + \mathbb {V} _ {2 2} (t) (u _ {2} ^ {1} (t) - F _ {2} (t) x _ {u} ^ {1} (t)) | ^ {2} \right. \right. \\ \left. - \left| \mathbb {V} _ {1 1} (t) (u _ {1} ^ {1} (t) - F _ {1} (t) x _ {u} ^ {1} (t)) | ^ {2}\right) d t \right] \\ + \frac {1}{2} \mathbb {E} \Big [ \int_ {s} ^ {T} \big (| \hat {\mathbb {V}} _ {2 1} (t) (u _ {1} ^ {2} (t) - \hat {F} _ {1} (t) x _ {u ^ {2}} ^ {2} (t)) + \hat {\mathbb {V}} _ {2 2} (t) (u _ {2} ^ {2} (t) - \hat {F} _ {2} (t) x _ {u ^ {2}} ^ {2} (t)) | ^ {2} \\ \end{array}
\left. - \left| \hat {\mathbb {V}} _ {1 1} (t) \left(u _ {1} ^ {2} (t) - \hat {F} _ {1} (t) x _ {u ^ {2}} ^ {2} (t)\right) \right| ^ {2}\right) d t \Bigg ] \tag {42}$$

It follows from the uniqueness of the solution of an initial value problem (IVP) that:

$$
\begin{array}{l} \inf _ {u _ {2} (\cdot) \in \mathcal {U} _ {2}} \mathcal {J} (s, T, x _ {s}; u _ {1} (\cdot), u _ {2} (\cdot)) = \mathcal {J} (s, T, x _ {s}; u _ {1} (\cdot), \tilde {u} _ {2} (\cdot , u _ {1} (\cdot))) \\ = \frac {1}{2} \mathbb {E} [ \langle \hat {X} (s) x _ {s}, x _ {s} \rangle ] - \frac {1}{2} \mathbb {E} \left[ \int_ {s} ^ {T} | \mathbb {V} _ {1 1} (t) (u _ {1} ^ {1} (t) - F _ {1} (t) x _ {u _ {1}} ^ {1} (t)) | ^ {2} d t \right] \\ - \frac {1}{2} \mathbb {E} \left[ \int_ {s} ^ {T} \left| \hat {\mathbb {V}} _ {1 1} (t) \left(u _ {1} ^ {2} (t) - \hat {F} _ {1} (t) x _ {u _ {1} ^ {2}} ^ {2} (t)\right) \right| ^ {2} d t \right] \tag {43} \\ \end{array}
$$

for all $u _ { 1 } ( \cdot ) \in L _ { w } ^ { 2 } ( [ s , T ] , \mathbb { R } ^ { m _ { 1 } } )$ . From (43) and (41) we obtain that:

$$\sup _ {u _ {1} (\cdot) \in \mathcal {U} _ {1}} \inf _ {u _ {2} (\cdot) \in \mathcal {U} _ {2}} \mathcal {J} (s, T, x _ {s}; u _ {1} (\cdot), u _ {2} (\cdot)) = \mathcal {J} (s, T, x _ {s}; \tilde {u} _ {1} (\cdot), \tilde {u} _ {2} (\cdot , \tilde {u} _ {1} (\cdot)))$$

which confirms the validity of the first equality from (6). On the other hand, from Lemma 5.1 we obtain that

$$\mathcal {J} (s, T, x _ {s}; \tilde {u} _ {1} (\cdot), \tilde {u} _ {2} (\cdot , \tilde {u} _ {1} (\cdot))) \leq \mathcal {J} (s, T, x _ {s}; \tilde {u} _ {1} (\cdot), u _ {2} (\cdot))$$

for all $u _ { 2 } ( \cdot ) \in L _ { w } ^ { 2 } ( [ s , T ] ; \mathbb { R } ^ { m _ { 2 } } )$ . Hence,

$$\mathcal {J} (s, T, x _ {s}; \tilde {u} _ {1} (\cdot), \tilde {u} _ {2} (\cdot , \tilde {u} _ {1} (\cdot))) = \inf _ {u _ {2} (\cdot) \in \mathcal {U} _ {2}} \mathcal {J} (s, T, x _ {s}; \tilde {u} _ {1} (\cdot), u _ {2} (\cdot))$$

because $\tilde { u } _ { 2 } ( \cdot , \tilde { u } _ { 1 } ( \cdot ) ) \in \mathcal { U } _ { 2 }$ . So we have shown that the second equality from (6) holds true. Thus the proof is complete.
