$$\mathbb {E} [ V (t + h, \mathbf {X} _ {t + h}) \mid \mathbf {X} _ {\mathbf {t}} = \mathbf {x} ] = \int_ {\mathbb {R} _ {+} ^ {n}} V (t + h, \mathbf {y}) f (\mathbf {y} \mid \mathbf {x}) d \mathbf {y}= \sum_ {k _ {1}, \ldots , k _ {n} = 0} ^ {\infty} \left(\int_ {\mathbb {R} _ {+} ^ {n}} V (t + h, \mathbf {y}) \prod_ {i = 1} ^ {d} g _ {k _ {i}} ^ {(i)} (y _ {i}) d \mathbf {y}\right) \mathbb {E} _ {\mathbf {x}} \left[ \prod_ {i = 1} ^ {d} g _ {k _ {i}} ^ {(i)} (X _ {t + h} ^ {(i)}) \right]= \sum_ {k _ {1}, \ldots , k _ {n} = 0} ^ {\infty} \left(\int_ {\mathbb {R} _ {+} ^ {n}} V (t + h, \mathbf {y}) \prod_ {i = 1} ^ {d} g _ {k _ {i}} ^ {(i)} (y _ {i}) d \mathbf {y}\right) \left(\prod_ {i = 1} ^ {d} g _ {k _ {i}} ^ {i} (x ^ {i}) + \int_ {t} ^ {t + h} \mathbb {E} _ {\mathbf {x}} \left[ \mathcal {L} ^ {\pi} \left(\prod_ {i = 1} ^ {d} g _ {k _ {i}} ^ {(i)} (X _ {u} ^ {(i)})\right) \right] d u\right),$$

$$\int_ {t} ^ {t + h} \mathbb {E} _ {\mathbf {x}} \left[ \mathcal {L} ^ {\pi} \left(\prod_ {i = 1} ^ {d} g _ {k _ {i}} ^ {(i)} (X _ {u} ^ {(i)})\right) \right] d u \approx h \cdot \mathcal {L} ^ {\pi} \left(\prod_ {i = 1} ^ {d} g _ {k _ {i}} ^ {(i)} (x _ {i})\right).$$
