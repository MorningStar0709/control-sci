$$
\begin{array}{l} \mathbb {E} _ {(x, i)} \left[ U (X _ {T \wedge \eta_ {n}} ^ {\xi}, Y _ {T \wedge \eta_ {n}}; \theta) \right] \leq U (x, i; \theta) + \mathbb {E} _ {(x, i)} \left[ \int_ {0} ^ {T \wedge \eta_ {n}} \left(\lambda (\theta , Y _ {s}) - \pi (X _ {s} ^ {\xi}, \theta)\right) d s \right] \tag {3.23} \\ + \mathbb {E} _ {(x, i)} \bigg [ \int_ {0} ^ {T \wedge \eta_ {n}} U _ {x} (X _ {s} ^ {\xi}, Y _ {s}; \theta) (d \xi_ {s} ^ {+, c} - d \xi_ {s} ^ {-, c}) \bigg ] \\ + \mathbb {E} _ {(x, i)} \bigg [ \sum_ {0 \leq s \leq T \land \eta_ {n}, s \in \Lambda^ {\xi}} \big (U (X _ {s +} ^ {\xi}, Y _ {s}; \theta) - U (X _ {s} ^ {\xi}, Y _ {s}; \theta) \big) \bigg ]. \\ \end{array}
$$

Noticing that

$$
\begin{array}{l} U (X _ {s ^ {+}} ^ {\xi}, Y _ {s}; \theta) - U (X _ {s} ^ {\xi}, Y _ {s}; \theta) = \mathbf {1} _ {\{\Delta \xi_ {s} ^ {+} > 0 \}} \int_ {0} ^ {\Delta \xi_ {s} ^ {+}} U _ {x} (X _ {s} ^ {\xi} + z, Y _ {s}; \theta) d z \\ - \mathbf {1} _ {\{\Delta \xi_ {s} ^ {-} > 0 \}} \int_ {0} ^ {\Delta \xi_ {s} ^ {-}} U _ {x} (X _ {s} ^ {\xi} - z, Y _ {s}; \theta) d z, \\ \end{array}
$$

using that $k _ { 2 } \leq U _ { x } \leq k _ { 1 }$ (cf. fourth and sixth conditions in (3.12)) in the latter display equation, and then using the obtained inequality in (3.23), we obtain

$$
\begin{array}{l} \mathbb {E} _ {(x, i)} \left[ U (X _ {T \wedge \eta_ {n}} ^ {\xi}, Y _ {T \wedge \eta_ {n}}; \theta) \right] \leq U (x, i; \theta) + \mathbb {E} _ {(x, i)} \left[ \int_ {0} ^ {T \wedge \eta_ {n}} \left(\lambda (\theta , Y _ {s}) - \pi (X _ {s} ^ {\xi}, \theta)\right) d s \right] \tag {3.24} \\ + \mathbb {E} _ {(x, i)} \bigg [ \int_ {0} ^ {T \wedge \eta_ {n}} (k _ {1} d \xi_ {t} ^ {+} - k _ {2} d \xi_ {t} ^ {-}) \bigg ]. \\ \end{array}
$$

$$\mathbb {E} _ {(x, i)} \left[ U (X _ {T} ^ {\xi}, Y _ {T}; \theta) \right] \leq U (x, i; \theta) + \mathbb {E} _ {(x, i)} \bigg [ \int_ {0} ^ {T} \bigg (\lambda (\theta , Y _ {s}) - \pi (X _ {s} ^ {\xi}, \theta) \bigg) d s + k _ {1} \xi_ {T} ^ {+} - k _ {2} \xi_ {T} ^ {-} \bigg ]. \tag {3.25}$$
