$$
\begin{array}{l} \lim _ {t \rightarrow \infty} \left(\exp (\xi_ {i}) | \zeta_ {i} | - \frac {\sigma_ {\max} (\mathcal {L} + \mathcal {G})}{\sigma_ {\min} (\mathcal {L} + \mathcal {G}) - \frac {1}{4}} | \delta_ {i} | \right. \tag {36} \\ \left. - \frac {\dot {\xi} _ {i} ^ {2} | \zeta_ {i} |}{(4 \sigma_ {\min} (\mathcal {L} + \mathcal {G}) - 1) \exp (\xi_ {i})}\right) \geq 0 \\ \end{array}
$$

Since singular values are non-negative real numbers, based on (36), one has

$$
\begin{array}{l} \lim _ {t \rightarrow \infty} \left(\| \operatorname{diag} \left(\exp \left(\xi_ {i}\right)\right) \zeta \| - \frac {\sigma_ {\max} (\mathcal {L} + \mathcal {G})}{\sigma_ {\min} (\mathcal {L} + \mathcal {G}) - \frac {1}{4}} \| \delta \| \right. \\ \left. - \frac {\left\| \operatorname{diag} \left(\dot {\xi} _ {i}\right) \zeta \right\| ^ {2}}{\left(4 \sigma_ {\min} (\mathcal {L} + \mathcal {G}) - 1\right) \| \operatorname{diag} \left(\exp \left(\xi_ {i}\right)\right) \zeta \|}\right) \geq 0 \tag {37} \\ \end{array}
$$
