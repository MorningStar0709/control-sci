$$
\begin{array}{l} - \Delta \tilde {e} _ {y} + \lambda^ {- 1} \tilde {e} _ {p} = f - \lambda^ {- 1} p _ {\sigma} + \Delta y _ {\theta} + \lambda^ {- 1} \zeta^ {p} \text {in} \Omega \quad \text {with} \quad \tilde {e} _ {y} = 0 \text {on} \partial \Omega , \\ - \Delta \tilde {e} _ {p} - \tilde {e} _ {y} = y _ {\theta} - y _ {d} + \Delta p _ {\sigma} - \zeta^ {y} \mathrm{in} \Omega \qquad \mathrm{with} \quad \tilde {e} _ {p} = 0 \mathrm{on} \partial \Omega . \\ \end{array}
$$

$$
\begin{array}{l} \| \nabla \tilde {e} _ {y} \| _ {L ^ {2} (\Omega)} ^ {2} + \lambda^ {- 1} \| \nabla \tilde {e} _ {p} \| _ {L ^ {2} (\Omega)} ^ {2} \leq \| f - \lambda^ {- 1} p _ {\sigma} + \Delta y _ {\theta} + \lambda^ {- 1} \zeta^ {p} \| _ {L ^ {2} (\Omega)} \| \tilde {e} _ {y} \| _ {L ^ {2} (\Omega)} \\ + \lambda^ {- 1} \| y _ {\theta} - y _ {d} + \Delta p _ {\sigma} - \zeta^ {y} \| _ {L ^ {2} (\Omega)} \| \tilde {e} _ {p} \| _ {L ^ {2} (\Omega)}. \\ \end{array}
$$
