$$
\left\{ \begin{array}{l l} \langle \tilde {w} _ {N} (\cdot , t), \phi_ {j} \rangle_ {\mathcal {H}} + \int_ {0} ^ {t} \langle \tilde {w} _ {N x} (\cdot , \tau), \phi_ {j x} \rangle_ {\mathcal {H}} d \tau = \langle \tilde {w} _ {N} (\cdot , 0), \phi_ {j} \rangle_ {\mathcal {H}} \\ - \phi_ {j} (1) \int_ {0} ^ {t} \left[ b \tilde {\zeta} _ {N} (\tau) u _ {0} (\tau) + c _ {1} \tilde {w} _ {N} (1, \tau) \right] d \tau , & j = 1, 2 \dots , N. \\ \tilde {\zeta} _ {N} (t) = \tilde {\zeta} (0) + \operatorname{sgn} (b) \int_ {0} ^ {t} \tilde {w} _ {N} (1, \tau) u _ {0} (\tau) d \tau , \end{array} \right. \tag {5.19}
$$

We pass to the limit as $N \to + \infty$ in (5.19) to get

$$
\left\{ \begin{array}{l} \langle \tilde {w} (\cdot , t), \phi_ {j} \rangle_ {\mathcal {H}} + \int_ {0} ^ {t} \langle \tilde {w} _ {x} (\cdot , \tau), \phi_ {j x} \rangle_ {\mathcal {H}} d \tau = \langle \tilde {w} (\cdot , 0), \phi_ {j} \rangle_ {\mathcal {H}} \\ - \phi_ {j} (1) \int_ {0} ^ {t} \left[ b \tilde {\zeta} (\tau) u _ {0} (\tau) + c _ {1} \tilde {w} (1, \tau) \right] d \tau , \\ \tilde {\zeta} (t) = \tilde {\zeta} (0) + \operatorname{sgn} (b) \int_ {0} ^ {t} \tilde {w} (1, \tau) u _ {0} (\tau) d \tau , \end{array} \right. \tag {5.20}
$$

which implies that $( \tilde { w } , \tilde { \zeta } )$ is a weak solution of the system (2.7). Since system (2.7) is a linear system, the uniqueness of the solution is trivial. By virtue of (5.18), we let $N \to + \infty$ in (5.13) and (5.12) to obtain (5.1) and (5.2), respectively. □
