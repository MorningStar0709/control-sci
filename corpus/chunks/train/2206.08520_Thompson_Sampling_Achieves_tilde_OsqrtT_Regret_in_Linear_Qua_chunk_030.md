$$
\begin{array}{l} L (A _ {c, *} + G _ {t}) = L (A _ {c, *}) + \nabla L (A _ {c, *}) \bullet G _ {t} + \frac {1}{2} G _ {t} \bullet \mathcal {H} _ {A _ {c, *} + s G _ {t}} (G _ {t}) \\ \leq L (A _ {c, *}) + \nabla L (A _ {c, *}) \bullet G _ {t} + \frac {r _ {*}}{2} \| G _ {t} \| _ {F} ^ {2} \tag {12} \\ \end{array}
$$

where $r _ { * } > 0$ is a constant due to Lemma 9. Using (12), we have the following lower bound on (10),

$$p _ {t} ^ {\text { opt }} \geq \min _ {\hat {\Upsilon}: \| \hat {\Upsilon} \| _ {F} \leq 1} \mathbb {P} _ {t} \left\{\frac {r _ {*}}{2} \| (\Xi + \hat {\Upsilon}) F _ {t} ^ {\frac {1}{2}} \| _ {F} ^ {2} + \nabla L _ {*} \bullet (\Xi + \hat {\Upsilon}) F _ {t} ^ {\frac {1}{2}} \leq 0, \text { and } \| (\Xi + \hat {\Upsilon}) F _ {t} ^ {\frac {1}{2}} \| _ {F} \leq \epsilon_ {*} \right\}, \tag {13}$$
