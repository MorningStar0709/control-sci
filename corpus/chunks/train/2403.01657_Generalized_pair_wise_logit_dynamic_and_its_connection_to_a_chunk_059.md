$$
\begin{array}{l} V _ {4} = - \left\{\int_ {A} \left(\int_ {\Omega} a ^ {(2)} (x, y, \mu^ {(1)}) \mathrm{d} y\right) \mu^ {(2)} (t, \mathrm{d} x) - \int_ {A} \left(\int_ {\Omega} a ^ {(2)} (x, y, \mu^ {(2)}) \mathrm{d} y\right) \mu^ {(2)} (t, \mathrm{d} x) \right\} \\ = \int_ {A} \left(\int_ {\Omega} a ^ {(2)} (x, y, \mu^ {(2)}) \mathrm{d} y\right) \mu^ {(2)} (t, \mathrm{d} x) - \int_ {A} \left(\int_ {\Omega} a ^ {(2)} (x, y, \mu^ {(1)}) \mathrm{d} y\right) \mu^ {(2)} (t, \mathrm{d} x). \tag {92} \\ = \int_ {A} \left(\int_ {\Omega} \left\{a ^ {(2)} (x, y, \mu^ {(2)}) - a ^ {(2)} (x, y, \mu^ {(1)}) \right\} d y\right) \mu^ {(2)} (t, d x) \\ \end{array}
$$

As in the earlier cases, we obtain

$$V _ {4} \leq L _ {a} \left\| \mu^ {(1)} (t, \cdot) - \mu^ {(2)} (t, \cdot) \right\|. \tag {93}$$

Consequently, by (83), (88), (89), (91), and (93), we obtain

$$
\begin{array}{l} \frac {\mathrm{d} \mu^ {(1)} (t , A)}{\mathrm{d} t} - \frac {\mathrm{d} \mu^ {(2)} (t , A)}{\mathrm{d} t} \leq L _ {a} \| \mu^ {(1)} (t, \cdot) - \mu^ {(2)} (t, \cdot) \| + C _ {M} \varpi \left(\left| \kappa_ {1} - \kappa_ {2} \right|\right) + \left\| \mu^ {(1)} (t, \cdot) - \mu^ {(2)} (t, \cdot) \right\| \\ + \left\| \mu^ {(1)} (t, \cdot) - \mu^ {(2)} (t, \cdot) \right\| + C _ {M} \varpi \left(\left| \kappa_ {1} - \kappa_ {2} \right|\right) + L _ {a} \left\| \mu^ {(1)} (t, \cdot) - \mu^ {(2)} (t, \cdot) \right\|. \tag {94} \\ = 2 \left(\left(L _ {a} + 1\right) \left\| \mu^ {(1)} (t, \cdot) - \mu^ {(2)} (t, \cdot) \right\| + C _ {M} \varpi \left(\left| \kappa_ {1} - \kappa_ {2} \right|\right)\right) \\ \end{array}
$$

Then, applying the classical Gronwall lemma to (94), we obtain

$$\left\| \mu^ {(1)} (t, \cdot) - \mu^ {(2)} (t, \cdot) \right\| \leq \tilde {C} (t) \varpi \left(\left| \kappa_ {1} - \kappa_ {2} \right|\right), t > 0, \tag {95}$$

where $\tilde { C } ( t )$ is positive and bounded at each $t \in \left[ 0 , T \right]$ , and is independent from $\kappa _ { 1 } , \kappa _ { 2 }$ . Here, we used

$$\left| \mu^ {(1)} (t, A) - \mu^ {(2)} (t, A) \right| \leq \int_ {0} ^ {t} 2 \left(\left(L _ {a} + 1\right) \| \mu^ {(1)} (s, \cdot) - \mu^ {(2)} (s, \cdot) \| + C _ {M} \varpi \left(\left| \kappa_ {1} - \kappa_ {2} \right|\right)\right) \tag {96}$$

and $\operatorname* { s u p } _ { A } \left. \mu ^ { ( 1 ) } \left( t , A \right) - \mu ^ { ( 2 ) } \left( t , A \right) \right. = \frac { 1 } { 2 } \left. \mu ^ { ( 1 ) } \left( t , \cdot \right) - \mu ^ { ( 2 ) } \left( t , \cdot \right) \right.$ because both $\mu ^ { ( 1 ) } , \mu ^ { ( 2 ) }$ are probability measures.

The desired result directly follows from (95).
