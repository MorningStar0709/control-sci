$$
\begin{array}{l} \sqrt {E [ | | x _ {c} (t _ {k}) | | ^ {2} + | | u _ {c} (t _ {k}) | | ^ {2} ]} \leq a \lambda^ {k} \sqrt {| | x _ {0} | | ^ {2} + | | u _ {- 1} | | ^ {2}} \\ (\forall k \in \mathbf {N} _ {0}, \forall x _ {0} \in \mathbf {R} ^ {n}, \forall u _ {- 1} \in \mathbf {R} ^ {m}). \tag {15} \\ \end{array}
