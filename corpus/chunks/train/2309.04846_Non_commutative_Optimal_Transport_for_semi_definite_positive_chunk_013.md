$$
\begin{array}{l} - \frac {\tau_ {2} \varepsilon}{\tau_ {2} + \varepsilon} \left[ \log \left(\frac {A}{\varepsilon \nu (Y)}\right) - \frac {\omega_ {2} - 2 | | c | | _ {\infty}}{\varepsilon} \right] \leq u ^ {(c, \tau_ {2}, \varepsilon)} \leq - \frac {\tau_ {2} \varepsilon}{\tau_ {2} + \varepsilon} \left[ \log (\mu (X)) + \frac {\omega_ {1} - | | c | | _ {\infty}}{\varepsilon} \right], \\ - \frac {\tau_ {1} \varepsilon}{\tau_ {1} + \varepsilon} \left[ \log \left(\frac {A}{\varepsilon \mu (X)}\right) - \frac {\omega_ {1} - 2 | | c | | _ {\infty}}{\varepsilon} \right] \leq v ^ {(c, \tau_ {1}, \varepsilon)} \leq - \frac {\tau_ {1} \varepsilon}{\tau_ {1} + \varepsilon} \left[ \log (\nu (Y)) + \frac {\omega_ {2} - | | c | | _ {\infty}}{\varepsilon} \right], \\ \end{array}
$$

which concludes the proof, as the boundary values are independent of the choice of u and v that satisfy (2.14). □
