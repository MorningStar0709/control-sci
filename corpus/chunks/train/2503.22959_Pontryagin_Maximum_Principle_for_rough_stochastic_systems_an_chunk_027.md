$$\| R _ {s, t} ^ {L} \| _ {m} \lesssim_ {\pmb {\eta}} (\| \delta \Delta z \| _ {\beta , m} + \sup _ {r \in [ 0, \tau ]} \| \Delta z _ {r} ^ {\prime} \| _ {m}) (t - s) ^ {\alpha + \beta} + (\| \mathbb {E}. R ^ {\Delta z} \| _ {\beta + \beta^ {\prime}, m} + \| \delta \Delta z ^ {\prime} \| _ {\beta^ {\prime}, m}) (t - s) ^ {\alpha + \beta + \beta^ {\prime}},\| \mathbb {E} _ {s} R _ {s, t} ^ {L} \| _ {m} \lesssim_ {\pmb {\eta}} (\| \mathbb {E}. R ^ {\Delta z} \| _ {\beta + \beta^ {\prime}, m} + \| \delta \Delta z ^ {\prime} \| _ {\beta^ {\prime}, m}) (t - s) ^ {\alpha + \beta + \beta^ {\prime}} + \sup _ {r \in [ 0, \tau ]} \| \Delta z _ {r} ^ {\prime} \| _ {m} (t - s) ^ {2 \alpha}.$$

Substituting (3.30), (3.28), (3.29), (3.31) into the above inequalities yields

$$
\begin{array}{l} \left\| R ^ {L} \right\| _ {\beta + \beta^ {\prime}, m} \lesssim_ {\boldsymbol {\eta}} K \left(\left\| \delta \Delta y \right\| _ {\beta , m} + \left\| \delta \Delta y ^ {\prime} \right\| _ {\beta^ {\prime}, m} + \left\| \mathbb {E}. R ^ {\Delta y} \right\| _ {\beta + \beta^ {\prime}, m}\right) \tau^ {\alpha - \beta^ {\prime}} \tag {3.33} \\ \| \mathbb {E}. R ^ {L} \| _ {\beta + \beta^ {\prime}, m} \lesssim_ {\boldsymbol {\eta}} K (\| \delta \Delta y \| _ {\beta , m} + \| \delta \Delta y ^ {\prime} \| _ {\beta^ {\prime}, m} + \| \mathbb {E}. R ^ {\Delta y} \| _ {\beta + \beta^ {\prime}, m}) \tau^ {2 \alpha - \beta - \beta^ {\prime}} \\ \end{array}
$$

and we conclude from (3.32) and (3.33) that

$$\| \Delta Y \| _ {\beta , m} \lesssim_ {\boldsymbol {\eta}} K \| (\Delta y, \Delta y ^ {\prime}) \| _ {\beta , \beta^ {\prime}, m} \tau^ {\frac {1}{2} - \beta}. \tag {3.34}$$

Since

$$R _ {s, t} ^ {\Delta Y} = \delta (\Delta Y) _ {s, t} - \Delta Y _ {s} ^ {\prime} \delta \eta_ {s, t} = \int_ {s} ^ {t} [ b _ {r} (y _ {r}) - b _ {r} (\bar {y} _ {r}) ] d r + \int_ {s} ^ {t} [ \sigma_ {r} (y _ {r}) - \sigma_ {r} (\bar {y} _ {r}) ] d W _ {r} + R _ {s, t} ^ {L}$$

it follows by (3.32) and (3.33) that

$$\| \mathbb {E}. R ^ {\Delta Y} \| _ {\beta + \beta^ {\prime}, m} \lesssim_ {\boldsymbol {\eta}} K \| (\Delta y, \Delta y ^ {\prime}) \| _ {\beta , \beta^ {\prime}, m} \tau^ {2 \alpha - \beta - \beta^ {\prime}}. \tag {3.35}$$

Moreover, in view of (3.34), (3.27) and (3.35),

$$\| (\Delta Y, \Delta Y ^ {\prime}) \| _ {\beta , \beta^ {\prime}, m} \leq C _ {\boldsymbol {\eta}, K} \| (\Delta y, \Delta y ^ {\prime}) \| _ {\beta , \beta^ {\prime}, m} \tau^ {(\frac {1}{2} - \beta) \wedge (\beta - \beta^ {\prime})}.$$
