$$\hat {X} ^ {\mathrm{T}} \boldsymbol {\Phi} ^ {\mathrm{T}} \boldsymbol {\Lambda} ^ {\mathrm{T}} (\boldsymbol {\Gamma} ^ {\mathrm{T}} \overline {{\boldsymbol {Q}}} \boldsymbol {\Gamma} + \overline {{\boldsymbol {R}}}) \boldsymbol {\Lambda} \boldsymbol {\Phi} \hat {\boldsymbol {X}} = \hat {X} ^ {\mathrm{T}} \boldsymbol {\Phi} ^ {\mathrm{T}} \boldsymbol {Q} \boldsymbol {\Gamma} (\boldsymbol {\Gamma} ^ {\mathrm{T}} \overline {{\boldsymbol {Q}}} \boldsymbol {\Gamma} + \overline {{\boldsymbol {R}}}) ^ {- 1} (\boldsymbol {\Gamma} ^ {\mathrm{T}} \boldsymbol {Q} \boldsymbol {\Gamma} + \overline {{\boldsymbol {R}}})\left(\boldsymbol {\Gamma} ^ {\mathrm{T}} \overline {{\boldsymbol {Q}}} \boldsymbol {\Gamma} + \overline {{\boldsymbol {R}}}\right) ^ {- 1} \boldsymbol {\Gamma} ^ {\mathrm{T}} \overline {{\boldsymbol {Q}}} \boldsymbol {\Phi} \hat {\boldsymbol {X}} = \hat {\boldsymbol {X}} ^ {\mathrm{T}} \boldsymbol {\Phi} ^ {\mathrm{T}} \overline {{\boldsymbol {Q}}} \boldsymbol {\Gamma} \left(\boldsymbol {\Gamma} ^ {\mathrm{T}} \overline {{\boldsymbol {Q}}} \boldsymbol {\Gamma} + \overline {{\boldsymbol {R}}}\right) ^ {- 1} \boldsymbol {\Gamma} ^ {\mathrm{T}} \boldsymbol {Q} \boldsymbol {\Phi} \hat {\boldsymbol {X}} = \hat {\boldsymbol {X}} ^ {\mathrm{T}} S \hat {\boldsymbol {X}} \tag {8-36}$$

而式(8-34)与式(8-36)相加,得

$$
\begin{array}{l} - 2 \boldsymbol {X} ^ {\mathrm{T}} \boldsymbol {S} \hat {\boldsymbol {X}} + \hat {\boldsymbol {X}} ^ {\mathrm{T}} \boldsymbol {S} \hat {\boldsymbol {X}} = (\hat {\boldsymbol {X}} - 2 \boldsymbol {X}) ^ {\mathrm{T}} \boldsymbol {S} \hat {\boldsymbol {X}} = - (\widetilde {\boldsymbol {X}} + \boldsymbol {X}) ^ {\mathrm{T}} \boldsymbol {S} (\boldsymbol {X} - \widetilde {\boldsymbol {X}}) \\ = - \widetilde {X} ^ {\mathrm{T}} S X + \widetilde {X} ^ {\mathrm{T}} S \widetilde {X} - X ^ {\mathrm{T}} S X + X ^ {\mathrm{T}} S \widetilde {X} \\ = \widetilde {\boldsymbol {X}} ^ {\mathrm{T}} \boldsymbol {S} \widetilde {\boldsymbol {X}} - \boldsymbol {X} ^ {\mathrm{T}} \boldsymbol {S} \boldsymbol {X} \tag {8-37} \\ \end{array}
$$
