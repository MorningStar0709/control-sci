$$
\begin{array}{l} L _ {X _ {\mathcal {N} ^ {\mathrm{SE}}}} ^ {\circ} V _ {\mathcal {N} ^ {\mathrm{SE}}} \left(\left\{x _ {i} \right\} _ {i \in \mathcal {N} ^ {\mathrm{SE}}}, t\right) \\ = \sup _ {\zeta \in \partial_ {x _ {\mathcal {N} ^ {\mathrm{SE}}}} ^ {\circ} V _ {\mathcal {N} ^ {\mathrm{SE}}}} \zeta \dot {x} _ {\mathcal {N} ^ {\mathrm{SE}}} \\ = \sup _ {\zeta \in \partial_ {x _ {\mathcal {N} ^ {\mathrm{SE}}}} ^ {\circ} V _ {\mathcal {N} ^ {\mathrm{SE}}}} \sum_ {i \in \mathcal {N} ^ {\mathrm{SE}}} \zeta_ {i} \dot {x} _ {i} \\ \leq \sum_ {i \in \mathcal {N} ^ {\mathrm{SE}}} \sup _ {\zeta \in \partial_ {x _ {\mathcal {N} ^ {\mathrm{SE}}}} ^ {\circ} V _ {\mathcal {N} ^ {\mathrm{SE}}}} \zeta_ {i} \dot {x} _ {i} = \sum_ {i \in \mathcal {N} ^ {\mathrm{SE}}} \sup _ {\zeta_ {i} \in \partial_ {x _ {i}} ^ {\circ} V _ {\mathcal {N} ^ {\mathrm{SE}}}} \zeta_ {i} \dot {x} _ {i}, \\ \end{array}
$$

where $\mathcal { N } ^ { \mathrm { S E } } ~ = ~ \{ i _ { 1 } , \dotsc , i _ { | \mathcal { N } ^ { \mathrm { S E } } | } \}$ with $| \mathcal { N } ^ { \mathrm { S E } } |$ denoting the cardinality of $\mathcal { N } ^ { \mathrm { S E } } , x _ { \mathcal { N } ^ { \mathrm { S E } } } = [ x _ { i _ { 1 } } ^ { \top } , \cdot . . , x _ { i _ { | \mathcal { N } ^ { \mathrm { S E } } | } } ^ { \top } ] ^ { \top }$ , and $X _ { \mathcal { N } ^ { \mathrm { S E } } } =$ $[ X _ { i _ { 1 } } ^ { \top } , \ldots , X _ { i _ { | N ^ { \mathrm { S E } } | } } ^ { \top } ] ^ { \top }$ |N. Hence, we can obtain

$$
\begin{array}{l} L _ {X _ {\mathcal {N} ^ {\mathrm{SE}}}} ^ {\circ} V _ {\mathcal {N} ^ {\mathrm{SE}}} \left(\left\{x _ {i} \right\} _ {i \in \mathcal {N} ^ {\mathrm{SE}}}, t\right) + \frac {\partial}{\partial t} V _ {\mathcal {N} ^ {\mathrm{SE}}} \left(\left\{x _ {i} \right\} _ {i \in \mathcal {N} ^ {\mathrm{SE}}}, t\right) \\ + \alpha (V _ {\mathcal {N} ^ {\mathrm{SE}}} (\{x _ {i} \} _ {i \in \mathcal {N} ^ {\mathrm{SE}}}, t)) \leq 0, \quad \forall t \geq 0. \\ \end{array}
$$

By Theorem III.1, this concludes the proof.
