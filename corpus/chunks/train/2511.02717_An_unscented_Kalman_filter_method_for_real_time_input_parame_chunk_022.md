$$

$$
\left[ \begin{array}{c} \ddot {x} _ {1 (k)} ^ {m} \\ \ddot {x} _ {2 (k)} ^ {m} \end{array} \right] = \left[ \mathbf {M} ^ {- 1} \left\{\mathbf {u} _ {\mathbf {k}} ^ {\mathrm{e}} - \mathbf {C} \dot {\mathbf {x}} _ {\mathbf {k}} ^ {\mathbf {p}} - \mathbf {K x} _ {\mathbf {k}} ^ {\mathbf {p}} - \mathbf {E} \left[ \begin{array}{c} \left(x _ {1 (k)} ^ {p}\right) ^ {3} \\ \left(x _ {2 (k)} ^ {p} - x _ {1 (k)} ^ {p}\right) ^ {3} \end{array} \right] \right\} \right] \tag {27}
