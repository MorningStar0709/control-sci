$$x (t) = 1 - \frac {1}{2 \sqrt {\zeta^ {2} - 1} (\zeta - \sqrt {\zeta^ {2} - 1})} \mathrm{e} ^ {(- \zeta \omega_ {\mathrm{n}} + \omega_ {\mathrm{n}} \sqrt {\zeta^ {2} - 1}) t} +\frac {1}{2 \sqrt {\zeta^ {2} - 1} (\zeta + \sqrt {\zeta^ {2} - 1})} \mathrm{e} ^ {(- \zeta \omega_ {\mathrm{n}} - \omega_ {\mathrm{n}} \sqrt {\zeta^ {2} - 1}) t} \tag {5.3.15}$$
