$$e _ {w} (E _ {R}, e _ {\Omega}) = \frac {\mathrm{d}}{\mathrm{d} t} e _ {R} = \sum_ {i = 1} ^ {3} K _ {i} \mathbf {e} _ {i} \times (e _ {\Omega} \times E _ {R} ^ {\mathrm{T}} \mathbf {e} _ {i}). \tag {66}$$

$$
\begin{array}{l} \dot {\hat {R}} = \hat {R} \hat {\Omega} ^ {\times}, \\ \dot {\hat {\Omega}} = E _ {R} J ^ {- 1} \Big [ J \Omega \times \Omega + \widehat {\tau} _ {D} + \tau + k _ {a 1} J \phi_ {1} (\psi_ {a}) \Big ] \\ + E _ {R} J ^ {- 1} \left[ \kappa_ {a} J \left(e _ {R} ^ {\mathrm{T}} e _ {R}\right) ^ {\frac {1 - p}{p}} H \left(e _ {R}, \frac {p - 1}{p}\right) e _ {w} \right] \tag {67} \\ + E _ {R} J ^ {- 1} (\kappa_ {a} J e _ {w}) + E _ {R} e _ {\Omega} ^ {\times} E _ {R} ^ {\mathrm{T}} \widehat {\Omega}, \\ \dot {\hat {\tau}} _ {D} = J k _ {a 2} \phi_ {2} (\psi_ {a}), \\ \end{array}
