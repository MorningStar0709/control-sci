$$
\left\{ \begin{array}{l} \langle \tilde {w} _ {N} (\cdot , t), \phi_ {j} \rangle_ {\mathcal {H}} + \int_ {0} ^ {t} \langle \tilde {w} _ {N x} (\cdot , \tau), \phi_ {j x} \rangle_ {\mathcal {H}} d \tau = \langle \tilde {w} _ {N} (\cdot , 0), \phi_ {j} \rangle_ {\mathcal {H}} \\ + \phi_ {j} (1) \int_ {0} ^ {t} \left[ b (q + c _ {0}) \tilde {\zeta} _ {N} (\tau) \check {w} _ {N} (1, \tau) - c _ {1} \tilde {w} _ {N} (1, \tau) \right] d \tau , \\ \tilde {\zeta} _ {N} (t) = \tilde {\zeta} (0) - \operatorname{sgn} (b) (q + c _ {0}) \int_ {0} ^ {t} \tilde {w} _ {N} (1, \tau) \check {w} _ {N} (1, \tau) d \tau , \\ \langle \check {w} _ {N} (\cdot , t), \phi_ {j} \rangle_ {\mathcal {H}} + \int_ {0} ^ {t} \langle \check {w} _ {N x} (\cdot , \tau), \phi_ {j x} \rangle_ {\mathcal {H}} d \tau = \langle \check {w} _ {N} (\cdot , 0), \phi_ {j} \rangle_ {\mathcal {H}} \\ + \phi_ {j} (1) \int_ {0} ^ {t} [ c _ {1} \tilde {w} _ {N} (1, \tau) - c _ {0} \check {w} _ {N} (1, \tau) ] d \tau . \end{array} \right. \tag {6.33}
$$

In view of (6.15), (6.16), (6.17), (6.31), (6.32), (6.5) and (6.6), we pass to the limit as $N \to + \infty$ in (6.33) to get

$$
\left\{ \begin{array}{l l} \langle \tilde {w} (\cdot , t), \phi_ {j} \rangle_ {\mathcal {H}} + \int_ {0} ^ {t} \langle \tilde {w} _ {x} (\cdot , \tau), \phi_ {j x} \rangle_ {\mathcal {H}} d \tau = \langle \tilde {w} (\cdot , 0), \phi_ {j} \rangle_ {\mathcal {H}} \\ + \phi_ {j} (1) \int_ {0} ^ {t} \left[ b (q + c _ {0}) \tilde {\zeta} (\tau) \check {w} (1, \tau) - c _ {1} \tilde {w} (1, \tau) \right] d \tau , \\ \tilde {\zeta} (t) = \tilde {\zeta} (0) - \operatorname{sgn} (b) (q + c _ {0}) \int_ {0} ^ {t} \tilde {w} (1, \tau) \check {w} (1, \tau) d \tau , & j = 1, 2 \dots , \\ \langle \check {w} (\cdot , t), \phi_ {j} \rangle_ {\mathcal {H}} + \int_ {0} ^ {t} \langle \check {w} _ {x} (\cdot , \tau), \phi_ {j x} \rangle_ {\mathcal {H}} d \tau = \langle \check {w} (\cdot , 0), \phi_ {j} \rangle_ {\mathcal {H}} \\ + \phi_ {j} (1) \int_ {0} ^ {t} [ c _ {1} \tilde {w} (1, \tau) - c _ {0} \check {w} (1, \tau) ] d \tau , & \end{array} \right. \tag {6.34}
$$

which implies that $( \tilde { w } , \tilde { \zeta } , \check { w } )$ is a weak solution of the system (6.1). Since the weak convergence (6.15) and (6.16), (6.2) follows from (6.12), (6.6) and the weakly lower semicontinuity of the norm $\| \cdot \| _ { L ^ { 2 } ( 0 , \infty ) } \ [ 3 , \mathrm { p . 6 } $ , Theorem 1.1.1]. □
