$$\| \gamma^ {*, N} (z, \nu) - \gamma^ {*, N} (x, \mu) \| ^ {2}\leq \frac {2}{\rho} \langle \nabla_ {a} F _ {N} (z, \gamma^ {*, N} (z, \nu), \nu) - \nabla_ {a} F _ {N} (x, \gamma^ {*, N} (z, \nu), \mu), \gamma^ {*, N} (x, \mu) - \gamma^ {*, N} (z, \nu) \rangle\leq \frac {2}{\rho} \| \nabla_ {a} F _ {N} (z, \gamma^ {*, N} (z, \nu), \nu) - \nabla_ {a} F _ {N} (x, \gamma^ {*, N} (z, \nu), \mu) \| \| \gamma^ {*, N} (z, \nu) - \gamma^ {*, N} (x, \mu) \|\leq \left\{\frac {2 \mathbb {L} _ {1} (\mathsf {R} _ {1} ^ {N} , \mathsf {R} _ {2} ^ {N} , \mathsf {R} _ {3} ^ {N})}{\rho} d _ {\mathsf {X}} (x, z) + \frac {2 \mathbb {L} _ {2} (\mathsf {R} _ {1} ^ {N} , \mathsf {R} _ {2} ^ {N} , \mathsf {R} _ {3} ^ {N})}{\rho} W _ {1} (\mu , \nu) \right\} \| \gamma^ {*, N} (z, \nu) - \gamma^ {*, N} (x, \mu) \|$$

(by property (F2))

This implies that

$$
\begin{array}{l} \| \gamma^ {*, N} (z, \nu) - \gamma^ {*, N} (x, \mu) \| \leq \left\{\frac {2 \mathbb {L} _ {1} (\mathsf {R} _ {1} ^ {N} , \mathsf {R} _ {2} ^ {N} , \mathsf {R} _ {3} ^ {N})}{\rho} d _ {\mathsf {X}} (x, z) + \frac {2 \mathbb {L} _ {2} (\mathsf {R} _ {1} ^ {N} , \mathsf {R} _ {2} ^ {N} , \mathsf {R} _ {3} ^ {N})}{\rho} W _ {1} (\mu , \nu) \right\} \\ =: \mathsf {L} _ {1} ^ {*, N} d _ {\mathsf {X}} (x, z) + \mathsf {L} _ {2} ^ {*, N} W _ {1} (\mu , \nu) \\ \end{array}
$$

for all $( \mu , x ) , ( \nu , z ) \in \mathcal { P } ( \mathsf { X } ) \times \mathsf { X }$ , which completes the proof.
