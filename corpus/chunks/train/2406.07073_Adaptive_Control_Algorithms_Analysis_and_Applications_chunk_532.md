$$y (t + d + 1) = \theta_ {C} ^ {T} \phi_ {C} (t) - C ^ {*} [ y (t + d) - \hat {y} ^ {0} (t + d) ] + C E e (t + d + 1) \tag {11.132}$$

$$
\begin{array}{l} \varepsilon^ {0} (t + d + 1) = y (t + d + 1) - \hat {y} ^ {0} (t + d + 1) \\ = [ \theta_ {C} - \hat {\theta} _ {C} (t) ] ^ {T} \phi_ {C} (t) - C ^ {*} \varepsilon^ {0} (t + d) + C E e (t + d + 1) \tag {11.133} \\ \end{array}
