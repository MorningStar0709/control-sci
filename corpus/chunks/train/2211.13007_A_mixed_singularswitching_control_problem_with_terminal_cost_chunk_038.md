$$
\begin{array}{l} \mathrm{e} ^ {- r (\hat {\tau} _ {i} ^ {q})} \hat {f} _ {\ell_ {i}} (X _ {\hat {\tau} _ {i} ^ {q}}, I _ {\hat {\tau} _ {i} ^ {q}}) - \mathrm{e} ^ {- r (\hat {\tau} _ {i + 1} ^ {q})} \hat {f} _ {\ell_ {i}} (X _ {\hat {\tau} _ {i + 1} ^ {q}}, I _ {\hat {\tau} _ {i + 1} ^ {q}}) \\ = \int_ {\hat {\tau} _ {i} ^ {q} +} ^ {\hat {\tau} _ {i + 1} ^ {q}} \mathrm{e} ^ {- r (s)} [ [ c (X _ {s}, I _ {s}) \hat {f} _ {\ell_ {i}} (X _ {s}, I _ {s}) - \mathcal {L} _ {\ell_ {i}, I _ {s}} \hat {f} _ {\ell_ {i}} (X _ {s}, I _ {s}) ] \mathrm{d} s + \langle \mathrm{D} ^ {1} \hat {f} _ {\ell_ {i}} (X _ {s}, I _ {s}), \mathfrak {n} _ {s} \rangle \mathrm{d} \zeta_ {s} ^ {\mathrm{c}} ] \\ - \widetilde {\mathcal {M}} [ \hat {\tau} _ {i} ^ {q}, \hat {\tau} _ {i + 1} ^ {q}; X, I, \hat {f} _ {\ell_ {i}} ] - \sum_ {\hat {\tau} _ {i} ^ {q} <   s \leq \hat {\tau} _ {i + 1} ^ {q}} \mathrm{e} ^ {- r (s)} \mathcal {J} [ X _ {s}, I _ {s}, \hat {f} _ {\ell_ {i}} ]. \tag {4.10} \\ \end{array}
$$

where the process

$$
\begin{array}{l} \widetilde {\mathcal {M}} [ \hat {\tau} _ {i} ^ {q}, t \wedge \hat {\tau} _ {i + 1} ^ {q}; X, I, \hat {f} _ {\ell_ {i}} ] := \int_ {\hat {\tau} _ {i} ^ {q} +} ^ {t \wedge \hat {\tau} _ {i + 1} ^ {q}} \mathrm{e} ^ {- r (s)} \langle \mathrm{D} ^ {1} \hat {f} _ {\ell_ {i}} (X _ {s}, I _ {s}), \sigma (X _ {s}, I _ {s}) \rangle \mathrm{d} W _ {s} \\ + \int_ {\hat {\tau} _ {i} ^ {q} +} ^ {t \wedge \hat {\tau} _ {i + 1} ^ {q}} \int_ {\mathbb {R}} \mathrm{e} ^ {- r (s)} [ \hat {f} _ {\ell_ {i}} (X _ {s -}, I _ {\hat {\tau} _ {i} ^ {q}} + \bar {h} _ {\ell_ {i}} (I _ {s -}, z)) - \hat {f} _ {\ell_ {i}} (X _ {s -}, I _ {s -}) ] [ N (\mathrm{d} s, \mathrm{d} z) - \mathrm{d} s \times \nu (\mathrm{d} z)) ] \\ \end{array}
$$

is a square-integrable martingale. Therefore, multiplying by $\mathbb { 1 } _ { \{ \tau _ { i } < \tau \} }$ and taking expected value in both sides of (4.10), we get (4.6). □
