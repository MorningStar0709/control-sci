$$
\begin{array}{l} \| \mathcal {A} _ {L} \mathcal {Z} _ {3} (I \otimes E ^ {\intercal}) \mathcal {A} _ {L} ^ {\intercal} \| \leq \| \mathcal {A} _ {L} \vec {\xi} \| \| \mathcal {A} _ {L} (I \otimes E) \vec {\omega} \| \\ \leq \kappa_ {\xi} \left[ \sum_ {i = 0} ^ {T} \| A _ {L} ^ {i} \| \right] \kappa_ {\omega} \| E \| \left[ \sum_ {i = 0} ^ {T} \| A _ {L} ^ {i} \| \right] \\ \leq \kappa_ {\xi} \kappa_ {\omega} \frac {C _ {L} ^ {2}}{[ 1 - \sqrt {\rho (A _ {L})} ] ^ {2}} \| E \| \\ \end{array}
$$

where the last equality follows by Lemma 6; second, similarly

$$
\begin{array}{l} \left\| \mathcal {A} _ {L} (I \otimes L) \vec {\omega} \vec {\omega} ^ {\intercal} (I \otimes E ^ {\intercal}) \mathcal {A} _ {L} ^ {\intercal} \right\| \\ \leq \left\| \mathcal {A} _ {L} (I \otimes L) \vec {\omega} \right\| \left\| \mathcal {A} _ {L} (I \otimes E) \vec {\omega} \right\| \\ \leq \kappa_ {\omega} ^ {2} \| L \| \| E \| \left[ \sum_ {i = 0} ^ {T} \| A _ {L} ^ {i} \| \right] ^ {2} \\ \leq \kappa_ {\omega} ^ {2} \frac {C _ {L} ^ {2}}{[ 1 - \sqrt {\rho (A _ {L})} ] ^ {2}} \| L \| \| E \|; \\ \end{array}
$$

and finally

$$
\begin{array}{l} \| \mathcal {A} _ {L} \mathcal {X} _ {L} \mathcal {M} _ {L} [ E ] ^ {\intercal} \| \leq \| \mathcal {A} _ {L} \vec {\eta} _ {L} \| \| \mathcal {M} _ {L} [ E ] \vec {\eta} _ {L} \| \\ \leq \kappa_ {L} ^ {2} \left[ \sum_ {i = 0} ^ {T} \| A _ {L} ^ {i} \| \right] \left[ \sum_ {i = 0} ^ {T} \| M _ {i} [ E ] \| \right] \\ \leq \kappa_ {L} ^ {2} \left[ \frac {C _ {L} + 2 C _ {L} ^ {3} \rho (A _ {L}) ^ {3 / 2}}{[ 1 - \sqrt {\rho (A _ {L})} ] ^ {3}} \right] \| E H \| \\ \end{array}
$$

where the last inequality follows by Lemma 6. Now, by combining the last three bounds we can claim that almost surely

$$\| S _ {L} (E, \mathcal {Y}) \| \leq \frac {\nu_ {L}}{\| H ^ {\intercal} H \| _ {*}} \| E \|.$$

This also implies that

$$\| \mathbb {E} \left[ S _ {L} (E, \mathcal {Y}) ^ {2} \right] \| \leq \mathbb {E} \left[ \| S _ {L} (E, \mathcal {Y}) \| ^ {2} \right] \leq \frac {\nu_ {L} ^ {2}}{\| H ^ {\intercal} H \| _ {*} ^ {2}} \| E \| ^ {2}.$$

Therefore, by Lemma 9 we obtain that

$$
\begin{array}{l} \mathbb {P} \left[ \| \frac {1}{M} \sum_ {i = 1} ^ {M} S _ {L} (E, \mathcal {Y} _ {T} ^ {i}) - \mathbb {E} \left[ S _ {L} (E, \mathcal {Y} _ {T} ^ {i}) \right] \| \geq t \right] \\ \leq 2 n \exp \left[ \frac {- M \| H ^ {\intercal} H \| _ {*} ^ {2} t ^ {2} / 2}{\nu_ {L} ^ {2} \| E \| ^ {2} + 2 \nu_ {L} \| H ^ {\intercal} H \| _ {*} \| E \| t / 3} \right] \\ \end{array}
$$

Thus, by substituting t with $t \| E \| / \| H ^ { \tau } H \|$ ∗ and applying this bound to (35) we obtain that

$$\mathbb {P} \left[ | \langle \nabla \widehat {J} _ {T} (L) - \nabla J _ {T} (L), E \rangle | \leq t \| E \| \right] \geq 1 - 2 n \exp \left[ \frac {- M t ^ {2} / 2}{\nu_ {L} ^ {2} + 2 \nu_ {L} t / 3} \right]$$

Finally, choosing $E = \nabla \widehat { J } _ { T } ( L ) - \nabla J _ { T } ( L )$ proves the second claim.
