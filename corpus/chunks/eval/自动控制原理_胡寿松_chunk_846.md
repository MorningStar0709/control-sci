(MATLAB)

MATLAB 文本：

G0=tf(1,[1,1,0]);
Gc=tf([16.5,215,1000],[0,1,0]);
Gp=tf(60.6,[1,13.03,60.6]);
sys1=tf(1000,[1,17.5,215,1000]); %定义闭环系统传递函数
sys2=Gc*Gp/(1+Gc*G0); %定义期望输入到被控对象输入之间的传递函数
figure(1)
step(sys1);
axis([0,1,0,1.4]);
grid;
figure(2)
step(sys2);
grid;
