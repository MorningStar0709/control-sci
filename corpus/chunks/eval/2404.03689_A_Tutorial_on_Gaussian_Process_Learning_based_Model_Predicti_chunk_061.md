$$\frac {\partial \mathbf {x} ^ {\top} \mathbf {a}}{\partial \mathbf {x}} = \frac {\partial \mathbf {a} ^ {\top} \mathbf {x}}{\partial \mathbf {x}} = \mathbf {a}, \tag {A.10}\frac {\partial \mathbf {x} ^ {\top} \mathbf {A x}}{\partial \mathbf {x}} = \left(\mathbf {A} + \mathbf {A} ^ {\top}\right) \mathbf {x}. \tag {A.11}$$
