$$\mathrm{RMSE} = \sqrt {\frac {1}{n N _ {M C} N _ {s t e p}} \sum_ {k = 1} ^ {N _ {M C}} \sum_ {j = 1} ^ {N _ {s t e p}} \sum_ {l = 1} ^ {n} (\hat {x} _ {k | k , j} (l) - x _ {k , j} (l)) ^ {2}},$$
