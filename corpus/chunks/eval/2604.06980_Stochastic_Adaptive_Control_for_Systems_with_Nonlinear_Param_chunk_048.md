$$
\begin{array}{l} \left\| x _ {t + h} ^ {*} - x _ {t + h} \right\| ^ {2} \leq 2 \left\| x _ {t + h} ^ {*} - \bar {x} _ {t + h} \right\| ^ {2} + 2 \left\| \bar {x} _ {t + h} - x _ {t + h} \right\| ^ {2} \\ \leq 2 \rho_ {1} \| x _ {t} ^ {*} - x _ {t} \| ^ {2} + 2 M _ {2} \sum_ {\tau = t} ^ {t + h} \| \theta^ {*} - \hat {\theta} _ {\tau - 1} \| ^ {2} \| x _ {\tau} \| ^ {2} + 2 M _ {2} t ^ {- 2 b}. \tag {D.3} \\ \end{array}
$$

Moreover, by the Cauchy–Schwarz inequality together with Theorems 2 and 3, we have

$$
\begin{array}{l} \sum_ {k = 0} ^ {\lfloor t / h \rfloor + 1} \sum_ {\tau = k h} ^ {k h + h} \| \theta^ {*} - \hat {\theta} _ {\tau - 1} \| ^ {2} \| x _ {\tau} \| ^ {2} \\ = O ((\sum_ {\tau = 0} ^ {t + h} \| \theta^ {*} - \hat {\theta} _ {\tau - 1} \| ^ {2 \frac {\gamma}{\gamma - 2}}) ^ {1 - \frac {2}{\gamma}} (\sum_ {\tau = 0} ^ {t + h} \| x _ {\tau} \| ^ {\gamma}) ^ {\frac {2}{\gamma}}) \tag {D.4} \\ = O ([ (\log t) ^ {1 + \delta} t ^ {(2 b + \eta)} ] ^ {1 - \frac {2}{\gamma}} t ^ {\frac {2}{\gamma}}), \text {   a.s. }, \\ \end{array}
$$

where $\lfloor t / h \rfloor$ denotes the integer part of $t / h$ . Then, combining (D.3) with (D.4), we get

$$
\begin{array}{l} \sum_ {\tau = 0} ^ {t} \left\| x _ {\tau} ^ {*} - x _ {\tau} \right\| ^ {2} \\ = O \left(\sum_ {k = 0} ^ {\lfloor t / h \rfloor + 1} \sum_ {\tau = k h} ^ {k h + h} \| \theta^ {*} - \hat {\theta} _ {\tau - 1} \| ^ {2} \| x _ {\tau} \| ^ {2}\right) + O \left(\sum_ {\tau = 0} ^ {t} \tau^ {- 2 b}\right) \\ = O ((\log t) ^ {\frac {(1 + \delta) (\gamma - 2)}{\gamma}} t ^ {(2 b + \eta - 1) (1 - \frac {2}{\gamma}) + 1} + t ^ {1 - 2 b}), a. s. \tag {D.5} \\ \end{array}
$$

Furthermore, note that

$$
\begin{array}{l} \left\| u _ {t} ^ {*} - u _ {t} \right\| ^ {2} \\ = \| \pi_ {\theta^ {*}} (x _ {t} ^ {*}) - \pi_ {\hat {\theta} _ {t - 1}} (x _ {t} ^ {*}) + \pi_ {\hat {\theta} _ {t - 1}} (x _ {t} ^ {*}) - \pi_ {\hat {\theta} _ {t - 1}} (x _ {t}) \| ^ {2} \\ = O \left(\| \theta^ {*} - \hat {\theta} _ {\tau - 1} \| ^ {2} \| x _ {t} ^ {*} \| ^ {2}\right) + O \left(\| x _ {t} - x _ {t} ^ {*} \| ^ {2}\right), a. s. \\ \end{array}
$$

Then, using (D.4) and (D.5), the following holds almost surely as $t \to \infty$ ,

$$
\begin{array}{l} \sum_ {\tau = 0} ^ {t} \left\| u _ {\tau} ^ {*} - u _ {\tau} \right\| ^ {2} \\ = O ((\log t) ^ {\frac {(1 + \delta) (\gamma - 2)}{\gamma}} t ^ {(2 b + \eta - 1) (1 - \frac {2}{\gamma}) + 1} + t ^ {1 - 2 b}). \tag {D.6} \\ \end{array}
$$

Theorem 4 then follows from (29), (D.5), and (D.6). ✷
