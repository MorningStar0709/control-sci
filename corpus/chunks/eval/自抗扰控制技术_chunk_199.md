$$
\left[ \begin{array}{l} \mathrm{fh} = - \dot {r} _ {0} ^ {2} \left(v _ {1} - v _ {0}\right) + 2 r _ {0} v _ {2} \\ \frac {\mathrm{d} v _ {1}}{\mathrm{d} \tau} = \rho v _ {2} \\ \frac {\mathrm{d} v _ {2}}{\mathrm{d} \tau} = \rho \mathrm{fh} \\ e = z _ {1} - y \\ \frac {\mathrm{d} z _ {1}}{\mathrm{d} \tau} = \rho \left(z _ {2} - \beta_ {0 1} e\right) \\ \frac {\mathrm{d} z _ {2}}{\mathrm{d} \tau} = \rho \left(z _ {3} - \beta_ {0 2} e + b _ {0} u\right) \\ \frac {\mathrm{d} z _ {3}}{\mathrm{d} \tau} = \rho (- \beta_ {0 3} e) \\ e _ {1} = v _ {1} - z _ {1}, e _ {2} = v _ {2} - z _ {2} \\ u = \frac {\beta_ {1} e _ {1} + \beta_ {2} e _ {2} - z _ {3}}{b _ {0}} \end{array} \right. \tag {5.6.9}
$$

再令

$$
\left\{ \begin{array}{l} v _ {1} ^ {\prime} = v _ {1}, v _ {2} ^ {\prime} = \rho v _ {2} \\ e ^ {\prime} = e, z _ {1} ^ {\prime} = z _ {1}, z _ {2} ^ {\prime} = \rho z _ {2}, z _ {3} ^ {\prime} = \rho^ {2} z _ {3}, u ^ {\prime} = \rho^ {2} u \\ e _ {1} ^ {\prime} = e _ {1}, e _ {2} ^ {\prime} = \rho e _ {2} \end{array} \right. \tag {5.6.10}

\left\{ \begin{array}{l} \mathrm{fh} ^ {\prime} = - \left(\rho r _ {0}\right) ^ {2} \left(v _ {1} ^ {\prime} - v _ {0}\right) + 2 \left(\rho r _ {0}\right) v _ {2} ^ {\prime} \\ \frac {\mathrm{d} v _ {1} ^ {\prime}}{\mathrm{d} \tau} = v _ {2} ^ {\prime} \\ \frac {\mathrm{d} v _ {2} ^ {\prime}}{\mathrm{d} \tau} = \mathrm{fh} ^ {\prime} \\ e ^ {\prime} = z _ {1} ^ {\prime} - y \\ \frac {\mathrm{dz} _ {1} ^ {\prime}}{\mathrm{d} \tau} = \left(z _ {2} ^ {\prime} - \rho \beta_ {0 1} e ^ {\prime}\right) \\ \frac {\mathrm{dz} _ {2} ^ {\prime}}{\mathrm{d} \tau} = \left(z _ {3} ^ {\prime} - \rho^ {2} \beta_ {0 2} e ^ {\prime} + b _ {0} u ^ {\prime}\right) \\ \frac {\mathrm{dz} _ {3} ^ {\prime}}{\mathrm{d} \tau} = \left(- \rho^ {3} \beta_ {0 3} e ^ {\prime}\right) \\ e _ {1} ^ {\prime} = v _ {1} ^ {\prime} - z _ {1}, e _ {2} ^ {\prime} = v _ {2} ^ {\prime} - z _ {2} ^ {\prime} \\ u ^ {\prime} = \frac {\rho^ {2} \beta_ {1} e _ {1} ^ {\prime} + \rho \beta_ {2} e _ {2} ^ {\prime} - z _ {3} ^ {\prime}}{b _ {0}} \end{array} \right. \tag {5.6.11}
