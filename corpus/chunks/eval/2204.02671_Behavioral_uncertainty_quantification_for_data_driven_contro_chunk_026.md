$$
\begin{array}{l} \lim _ {L \to \infty} \vec {\mathrm{gap}} _ {L} (\mathcal {B}, \tilde {\mathcal {B}}) \stackrel {(4)} {=} \lim _ {L \to \infty} \vec {\mathrm{gap}} (\mathcal {B} | _ {L}, \tilde {\mathcal {B}} | _ {L}) \\ \stackrel {(3)}{=}\lim_{L\to \infty}\sup_{\substack{w\in \mathcal{B}|_{L}\\ \| w\|_{2} = 1}}\inf_{\tilde{w}\in \tilde{\mathcal{B}} |_{L}}\| w - \tilde{w}\|_{2} \\ = \lim_{L\to \infty}\sup_{\substack{w\in \iota_{L}(\mathcal{B}|_{L})\\ \| w\|_{\ell_{2}} = 1}}\inf_{\tilde{w}\in \iota_{L}(\tilde{\mathcal{B}} |_{L})}\| w - \tilde{w}\|_{\ell_{2}} \\ = \lim_{L\to \infty}\sup_{\substack{w\in \Pi_{L}(\mathcal{B}))\\ \| w\|_{\ell_{2}} = 1}}\inf_{\tilde{w}\in \Pi_{L}(\tilde{\mathcal{B}})}\left\| w - \tilde{w}\right\|_{\ell_{2}} \\ \leq \lim_{L\to \infty}\sup_{\substack{w\in \Pi_{L}(\mathcal{B}))\\ \| w\|_{\ell_{2}} = 1}}\inf_{\tilde{w}\in \tilde{\mathcal{B}}}\| w - \tilde{w}\|_{\ell_{2}} \\ = \sup_{\substack{w\in \mathcal{B}\\ \| w\|_{\ell_{2}} = 1}}\inf_{\tilde{w}\in \tilde{\mathcal{B}}}\| w - \tilde{w}\|_{\ell_{2}} \\ = \vec {\operatorname{gap}} _ {\ell_ {2}} (P, \tilde {P}), \tag {12} \\ \end{array}
$$

where the third identity is a consequence of Lemma 4, the fourth identity follows from Lemma 6, the fifth inequality is implied by in $\begin{array} { r } { \mathrm { f } _ { _ { \tilde { w } \in \Pi _ { L } ( \tilde { \mathcal { B } } ) } } \| w - \tilde { w } \| _ { \ell _ { 2 } } \leq \operatorname* { i n f } _ { \tilde { w } \in \tilde { \mathcal { B } } } \| w - \tilde { w } \| _ { \ell _ { 2 } } } \end{array}$ for $w \in \Pi _ { L } ( B )$ , the sixth equality is a consequence of Lemma 5, and the last identity holds by Definition 9, since $B = \mathrm { g r a p h } _ { \ell _ { 2 } } ( P )$ and $B = \mathrm { g r a p h } _ { \ell _ { 2 } } ( \tilde { P } )$ , by assumption. Finally, the function $( x , y ) \mapsto \operatorname* { m a x } ( x , y )$ is continuous, thus

$$
\begin{array}{l} \lim _ {L \to \infty} \mathrm{gap} _ {L} (\mathcal {B}, \tilde {\mathcal {B}}) = \lim _ {L \to \infty} \max \{\overrightarrow {\mathrm{gap}} _ {L} (\mathcal {B}, \tilde {\mathcal {B}}), \overrightarrow {\mathrm{gap}} _ {L} (\tilde {\mathcal {B}}, \mathcal {B}) \} \\ \stackrel {(1 2)} {\leq} \max \{\vec {\operatorname{gap}} _ {\ell_ {2}} (P, \tilde {P}), \vec {\operatorname{gap}} _ {\ell_ {2}} (\tilde {P}, P) \} \\ = \operatorname{gap} _ {\ell_ {2}} (\tilde {P}, P). \\ \end{array}
$$
