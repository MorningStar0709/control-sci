$$
\begin{array}{l} V _ {a} \left(K _ {t + 1}\right) \leqslant V _ {a} \left(K _ {t}\right) + \frac {1}{\left\| \phi_ {j} \left(x _ {t}\right) \right\| ^ {2}} \operatorname{tr} \left(\theta \tilde {u} _ {t} \tilde {u} _ {t} ^ {\top} + 2 \left(- \tilde {u} _ {t} + \varepsilon_ {j} \left(x _ {t}\right)\right) \tilde {u} _ {t} ^ {\top}\right) \\ = V _ {a} \left(K _ {t}\right) + \frac {1}{\left\| \phi_ {j} \left(x _ {t}\right) \right\| ^ {2}} \left(\theta \| \tilde {u} _ {t} \| ^ {2} - 2 \tilde {u} _ {t} ^ {\top} \left(\tilde {u} _ {t} - \varepsilon_ {j} \left(x _ {t}\right)\right)\right) \\ = V _ {a} (K _ {t}) + \frac {1}{\left\| \phi_ {j} (x _ {t}) \right\| ^ {2}} \left((\theta - 2) \left\| \tilde {u} _ {t} \right\| ^ {2} + 2 \tilde {u} _ {t} ^ {\top} \varepsilon_ {j} (x _ {t})\right) \\ \leqslant V _ {a} \left(K _ {t}\right) + \frac {1}{\left\| \phi_ {j} \left(x _ {t}\right) \right\| ^ {2}} \left(\left(\theta - 1\right) \| \tilde {u} _ {t} \| ^ {2} + \left\| \varepsilon_ {j} \left(x _ {t}\right) \right\| ^ {2}\right) \\ = V _ {a} (K _ {t}) - \frac {1 - \theta}{\left\| \phi_ {j} (x _ {t}) \right\| ^ {2}} \left\| \tilde {u} _ {t} \right\| ^ {2} + \frac {1}{\left\| \phi_ {j} (x _ {t}) \right\| ^ {2}} \left\| \boldsymbol {\varepsilon} _ {j} (x _ {t}) \right\| ^ {2} \\ \leqslant V _ {a} \left(K _ {t}\right) - \frac {1 - \theta}{\sigma^ {2}} \left\| \tilde {u} _ {t} \right\| ^ {2} + \frac {1}{m ^ {2}} \left\| \varepsilon_ {j} \left(x _ {t}\right) \right\| ^ {2}, \\ \end{array}
$$

where the last inequality is due to $m ^ { 2 } \leqslant \left\| \phi _ { j } ( x _ { t } ) \right\| ^ { 2 } \leqslant \sigma ^ { 2 }$ . Therefore,

$$V _ {a} \left(K _ {t + 1}\right) - V _ {a} \left(K _ {t}\right) \leqslant - \frac {1 - \theta}{\sigma^ {2}} \left\| \tilde {u} _ {t} \right\| ^ {2} + \frac {1}{m ^ {2}} \left\| \varepsilon_ {j} \left(x _ {t}\right) \right\| ^ {2}.$$

(iii) Consider Lemma 1-(ii) to get

$$
\begin{array}{l} \frac {1 - \theta}{\sigma^ {2}} \| \tilde {u} _ {t} \| ^ {2} \leqslant - V _ {a} (K _ {t + 1}) + V _ {a} (K _ {t}) + \frac {1}{m ^ {2}} \left\| \varepsilon_ {j} (x _ {t}) \right\| ^ {2} \\ \leqslant - V _ {a} (K _ {t + 1}) + V _ {a} (K _ {t}) + \frac {\bar {\varepsilon} ^ {2}}{m ^ {2}}. \\ \end{array}
$$

By summing from t = k to $k + N - 1$ in both sides, we get

$$\frac {1 - \theta}{\sigma^ {2}} \sum_ {t = k} ^ {k + N - 1} \| \tilde {u} _ {t} \| ^ {2} \leqslant V _ {a} (K _ {k}) + \frac {N}{m ^ {2}} \bar {\varepsilon} ^ {2} \leqslant \frac {4}{\theta} \bar {W} + \frac {N}{m ^ {2}} \bar {\varepsilon} ^ {2}.$$
