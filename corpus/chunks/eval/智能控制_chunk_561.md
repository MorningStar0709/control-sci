$$\| \boldsymbol {e} _ {k + 1} (t) \| \leqslant \| \boldsymbol {I} - \boldsymbol {C} (t) \boldsymbol {B} (t) \boldsymbol {\Gamma} (t) \| \| \boldsymbol {e} _ {k} (t) \| + \int_ {0} ^ {t} \left| \left| \frac {\partial}{\partial \tau} \boldsymbol {G} (t, \tau) \right| \right| \| \boldsymbol {e} _ {k} (\tau) \| d \tau +\int_ {0} ^ {t} \| \boldsymbol {C} (t) \boldsymbol {\Phi} (t, \tau) \boldsymbol {B} (\tau) \boldsymbol {L} (\tau) \| \| \boldsymbol {e} _ {k} (\tau) \| d \tau +\int_ {0} ^ {t} \int_ {0} ^ {\tau} \| \boldsymbol {C} (t) \boldsymbol {\Phi} (t, \tau) \boldsymbol {B} (\tau) \psi (\tau) \| \| \boldsymbol {e} _ {k} (\sigma) \| d \sigma d \tau\leqslant \left\| \boldsymbol {I} - \boldsymbol {C} (t) \boldsymbol {B} (t) \boldsymbol {\Gamma} (t) \right\| \left\| \boldsymbol {e} _ {k} (t) \right\| + \int_ {0} ^ {t} b _ {1} \left\| \boldsymbol {e} _ {k} (\tau) \right\| d \tau + \int_ {0} ^ {t} \int_ {0} ^ {\tau} b _ {2} \left\| \boldsymbol {e} _ {k} (\sigma) \right\| d \sigma d \tau \tag {11.18}$$

式中

$$
\begin{array}{l} b _ {1} = \max _ {t, \tau \in [ 0, T ]} \left\{\left| \left| \frac {\partial}{\partial \tau} \mathbf {G} (t, \tau) \right| \right|, \| \mathbf {C} (t) \boldsymbol {\Phi} (t, \tau) \mathbf {B} (\tau) \mathbf {L} (\tau) \| \right\} \\ b _ {2} = \sup _ {t, \tau \in [ 0, T ]} \| \boldsymbol {C} (t) \boldsymbol {\Phi} (t, \tau) \boldsymbol {B} (\tau) \boldsymbol {\psi} (\tau) \| \\ \end{array}
$$

根据 $\lambda$ 范数的定义可知，函数 $f: [0, T] \to R^{n}$ 的 $\lambda$ 范数为 $\|f\|_{\lambda} = \sup_{0 \leqslant t \leqslant T} \{\|f(t)\|e^{-\lambda t}\}$ 。将式(11.18)两端同乘以 $\exp(-\lambda t), \lambda > 0$ ，并考虑到 $\int_{0}^{t} \exp(\lambda \tau) \, d\tau = \frac{\exp(\lambda t) - 1}{\lambda}$ ，则式(11.18)不等式右边第二项为

$$\exp (- \lambda t) \int_ {0} ^ {t} b _ {1} \| \boldsymbol {e} _ {k} (\tau) \| \mathrm{d} \tau = \exp (- \lambda t) \int_ {0} ^ {t} b _ {1} \| \boldsymbol {e} _ {k} (\tau) \| \exp (- \lambda \tau) \exp (\lambda \tau) \mathrm{d} \tau\leqslant b _ {1} \exp (- \lambda t) \| \boldsymbol {e} _ {k} (\tau) \| _ {\lambda} \int_ {0} ^ {t} \exp (\lambda \tau) d \tau= b _ {1} \exp (- \lambda t) \| e _ {k} (\tau) \| _ {\lambda} \frac {\exp (\lambda t) - 1}{\lambda}= \frac {b _ {1}}{\lambda} \| e _ {k} (\tau) \| _ {\lambda} \exp (- \lambda t) (\exp (\lambda t) - 1)= b _ {1} \frac {(1 - \exp (- \lambda t))}{\lambda} \| \boldsymbol {e} _ {k} (\tau) \| _ {\lambda}\leqslant b _ {1} \frac {(1 - \exp (- \lambda T))}{\lambda} \| e _ {k} (\tau) \| _ {\lambda} \tag {11.19}$$
