$$
\begin{array}{l} \frac {\partial \mathbf {J} (\delta \mathbf {u} _ {k})}{\partial \delta \mathbf {u} _ {k}} \approx \\ \frac {\partial \left((\mathbf {x} _ {d , k + 1} - \bar {\mathbf {x}} _ {k + 1} - \mathbf {H} ^ {\prime} \delta \mathbf {u} _ {k}) ^ {\top} \mathbf {Q} (\mathbf {x} _ {d , k + 1} - \bar {\mathbf {x}} _ {k + 1} - \mathbf {H} ^ {\prime} \delta \mathbf {u} _ {k})\right)}{\partial \delta \mathbf {u} _ {k}} + \frac {\partial \left((\bar {\mathbf {u}} _ {k} + \delta \mathbf {u} _ {k}) ^ {\top} \mathbf {R} (\bar {\mathbf {u}} _ {k} + \delta \mathbf {u} _ {k})\right)}{\partial \delta \mathbf {u} _ {k}}, \\ = \frac {\partial \left(\left(\tilde {\mathbf {x}} _ {k + 1} - \mathbf {H} ^ {\prime} \delta \mathbf {u} _ {k}\right) ^ {\top} \mathbf {Q} \left(\tilde {\mathbf {x}} _ {k + 1} - \mathbf {H} ^ {\prime} \delta \mathbf {u} _ {k}\right)\right)}{\partial \delta \mathbf {u}} + \frac {\partial \left(\left(\bar {\mathbf {u}} _ {k} + \delta \mathbf {u} _ {k}\right) ^ {\top} \mathbf {R} \left(\bar {\mathbf {u}} _ {k} + \delta \mathbf {u} _ {k}\right)\right)}{\partial \delta \mathbf {u}}, \\ = 2 \mathbf {Q} (\tilde {\mathbf {x}} _ {k + 1} - \mathbf {H} ^ {'} \delta \mathbf {u} _ {k}) (- \mathbf {H} ^ {\prime}) + 2 \mathbf {R} (\bar {\mathbf {u}} _ {k} + \delta \mathbf {u} _ {k}), \\ = 2 \left(- Q \tilde {\mathbf {x}} _ {k + 1} \mathbf {H} ^ {\prime} + Q \mathbf {H} ^ {\prime} \delta \mathbf {u} _ {k} \mathbf {H} ^ {\prime} + R \bar {\mathbf {u}} _ {k} + R \delta \mathbf {u} _ {k}\right), \\ = 2 \left(- \mathbf {H} ^ {' \top} \mathbf {Q} \tilde {\mathbf {x}} _ {k + 1} + \mathbf {H} ^ {' \top} \mathbf {Q} \mathbf {H} ^ {'} \delta \mathbf {u} _ {k} + \mathbf {R} \bar {\mathbf {u}} _ {k} + \mathbf {R} \delta \mathbf {u} _ {k}\right), \\ = 2 \left((\mathbf {H} ^ {' \top} \mathbf {Q} \mathbf {H} ^ {'} + \mathbf {R}) \delta \mathbf {u} _ {k} + \mathbf {R} \bar {\mathbf {u}} _ {k} - \mathbf {H} ^ {' \top} \mathbf {Q} \tilde {\mathbf {x}} _ {k + 1}\right), \\ = 0. \tag {A.13} \\ \end{array}
$$

This leads to the solution,

$$\delta \mathbf {u} _ {k} = \left(\mathbf {H} ^ {\prime \top} \mathbf {Q} \mathbf {H} ^ {\prime} + \mathbf {R}\right) ^ {- 1} \left(\mathbf {H} ^ {\prime \top} \mathbf {Q} \tilde {\mathbf {x}} _ {k + 1} - \mathbf {R} \bar {\mathbf {u}} _ {k}\right). \tag {A.14}$$
